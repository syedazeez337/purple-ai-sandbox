1// purple/src/sandbox/config.rs
2//!
3//! Sandbox configuration for path externalization
4//!
5//! This module provides configurable paths for the sandboxing system.
6//! All hardcoded paths have been externalized to allow for customization
7//! and better portability across different Linux distributions and setups.
8
9use once_cell::sync::Lazy;
10use serde::{Deserialize, Serialize};
11use std::path::{Path, PathBuf};
12use std::sync::Mutex;
13
14/// Sandbox configuration containing all configurable paths
15#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
16pub struct SandboxConfig {
17    /// Cgroup parent path (default: /sys/fs/cgroup/purple)
18    pub cgroup_parent_path: PathBuf,
19    /// iptables binary path (default: /usr/sbin/iptables)
20    pub iptables_path: PathBuf,
21    /// Sandbox root directory (default: /tmp/purple-sandbox)
22    pub sandbox_root: PathBuf,
23    /// Default scratch directory inside sandbox (default: /tmp)
24    pub default_scratch_dir: PathBuf,
25    /// Default audit log path (default: /tmp/audit.log)
26    pub default_audit_log: PathBuf,
27    /// proc filesystem path (default: /proc)
28    pub proc_path: PathBuf,
29    /// sys filesystem path (default: /sys)
30    pub sys_path: PathBuf,
31    /// Default PATH inside sandbox (default: /usr/local/sbin:/usr/local/bin:/usr/bin:/bin:/usr/sbin:/sbin)
32    pub sandbox_path: String,
33    /// Essential host paths to mount (default: /usr/bin, /bin, /usr/lib, /lib, /lib64)
34    pub essential_host_paths: Vec<PathBuf>,
35}
36
37impl Default for SandboxConfig {
38    fn default() -> Self {
39        Self {
40            cgroup_parent_path: PathBuf::from("/sys/fs/cgroup/purple"),
41            iptables_path: PathBuf::from("/usr/sbin/iptables"),
42            sandbox_root: PathBuf::from("/tmp/purple-sandbox"),
43            default_scratch_dir: PathBuf::from("/tmp"),
44            default_audit_log: PathBuf::from("/tmp/audit.log"),
45            proc_path: PathBuf::from("/proc"),
46            sys_path: PathBuf::from("/sys"),
47            sandbox_path: "/usr/local/sbin:/usr/local/bin:/usr/bin:/bin:/usr/sbin:/sbin"
48                .to_string(),
49            essential_host_paths: vec![
50                PathBuf::from("/usr/bin"),
51                PathBuf::from("/bin"),
52                PathBuf::from("/usr/lib"),
53                PathBuf::from("/lib"),
54                PathBuf::from("/lib64"),
55            ],
56        }
57    }
58}
59
60/// Global sandbox configuration - uses Lazy for thread-safe one-time initialization
61static SANDBOX_CONFIG: Lazy<Mutex<SandboxConfig>> =
62    Lazy::new(|| Mutex::new(SandboxConfig::default()));
63
64/// Set the global sandbox configuration (for testing or customization)
65pub fn set_config(config: SandboxConfig) {
66    let mut config_ref = SANDBOX_CONFIG.lock().unwrap();
67    *config_ref = config;
68}

127/// Generate a sandbox-specific root path
128#[inline]
129pub fn sandbox_root_for_id(sandbox_id: &str) -> PathBuf {
130    let config = SANDBOX_CONFIG.lock().unwrap();
131    config
132        .sandbox_root
133        .join(format!("purple-sandbox-{}", sandbox_id))
134}
135
136/// Generate cgroup path for a sandbox
137#[inline]
138pub fn cgroup_path_for_id(sandbox_id: &str) -> PathBuf {
139    let config = SANDBOX_CONFIG.lock().unwrap();
140    // The sandbox_id already contains the full identifier (e.g., "test-123")
141    // We need to produce "purple-sandbox-test-123" not "purple-test-123"
142    config
143        .cgroup_parent_path
144        .join(format!("purple-sandbox-{}", sandbox_id))
145}
