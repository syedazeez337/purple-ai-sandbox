34    /// Executes the agent's command within the sandboxed environment.
35    /// Returns the exit code of the sandboxed process.
36    pub fn execute(&self) -> Result<i32> {
37        log::info!("Sandbox: Executing agent command: {:?}", self.agent_command);
38        log::info!("Sandbox: Policy being applied: {:?}", self.policy.name);
39
40        // 1. Setup user namespace first (required for privilege management)
41        log::info!("Setting up user namespace...");
42        let (_sandbox_uid, _sandbox_gid) =
43            linux_namespaces::unshare_user_namespace().map_err(|e| {
44                PurpleError::NamespaceError(format!("User namespace setup failed: {}", e))
45            })?;
46
47        // 2. Setup network namespace (if isolated)
48        if self.policy.network.isolated {
49            log::info!("Setting up network namespace (isolated)...");
50            linux_namespaces::unshare_network_namespace().map_err(|e| {
51                PurpleError::NamespaceError(format!("Network namespace setup failed: {}", e))
52            })?;
53            self.apply_network_filtering()?;
54        } else {
55            log::info!("Network namespace not isolated by policy.");
56        }
57
58        // 3. Setup PID namespace (prepare for fork)
59        log::info!("Setting up PID namespace...");
60        linux_namespaces::unshare_pid_namespace().map_err(|e| {
61            PurpleError::NamespaceError(format!("PID namespace setup failed: {}", e))
62        })?;
63
64        // 4. Fork to enter the new PID namespace
65        log::info!("Forking to enter new PID namespace...");
66        match unsafe { fork() } {
67            Ok(ForkResult::Parent { child }) => {
68                log::info!("Parent: Waiting for child PID {}", child);
69
70                // Wait for the child to finish
71                let exit_code = match waitpid(child, None) {
72                    Ok(status) => {
73                        log::info!("Child exited with status: {:?}", status);
74                        match status {
75                            nix::sys::wait::WaitStatus::Exited(_, code) => code,
76                            nix::sys::wait::WaitStatus::Signaled(_, signal, _) => {
77                                128 + (signal as i32)
78                            }
79                            _ => 1, // Default error for other states
80                        }
81                    }
82                    Err(e) => {
83                        log::error!("Failed to wait for child: {}", e);
84                        return Err(PurpleError::CommandError(format!(
85                            "Failed to wait for child: {}",
86                            e
87                        )));
88                    }
89                };
90
91                // 8. Clean up and audit logging (Parent does cleanup)
92                self.cleanup_and_audit()?;
93
94                Ok(exit_code)
95            }
96            Ok(ForkResult::Child) => {
97                // Child Process: Setup environment and Exec
98                log::info!("Child: I am running in the new PID namespace!");
99
100                // 5. Mount namespace (Private)
101                if let Err(e) = linux_namespaces::unshare_mount_namespace() {
102                    log::error!("Mount namespace setup failed: {}", e);
103                    std::process::exit(1);
104                }
105
106                // 6. Setup Filesystem (Fresh proc!)
107                if let Err(e) = self.setup_filesystem() {
108                    log::error!("Filesystem setup failed: {}", e);
109                    std::process::exit(1);
110                }
111
112                // 7. Caps, Seccomp, Resources
113                if let Err(e) = self.apply_resource_limits() {
114                    log::error!("Resource limits failed: {}", e);
115                    std::process::exit(1);
116                }
117
118                if let Err(e) = self.drop_capabilities() {
119                    log::error!("Drop capabilities failed: {}", e);
120                    std::process::exit(1);
121                }
122
123                if let Err(e) = self.apply_syscall_filtering() {
124                    log::error!("Syscall filtering failed: {}", e);
125                    std::process::exit(1);
126                }
127
128                // 8. Exec
129                log::info!("Child: Executing actual command...");
130
131                let prog = CString::new(self.agent_command[0].clone()).unwrap();
132                let args: Vec<CString> = self
133                    .agent_command
134                    .iter()
135                    .map(|arg| CString::new(arg.clone()).unwrap())
136                    .collect();
137
138                // execvp never returns on success (it replaces the process),
139                // so we only need to handle the error case
140                #[allow(irrefutable_let_patterns)]
141                match execvp(&prog, &args) {
142                    Ok(_) => unreachable!("execvp should not return on success"),
143                    Err(e) => {
144                        log::error!("Failed to execvp: {}", e);
145                        std::process::exit(1);
146                    }
147                }
148            }
149            Err(e) => Err(PurpleError::NamespaceError(format!("Fork failed: {}", e))),
150        }
151    }

153    /// Sets up filesystem isolation with bind mounts and chroot
154    fn setup_filesystem(&self) -> Result<()> {
155        log::info!("Setting up filesystem isolation...");
156
157        // Create temporary directory structure for the sandbox
158        let sandbox_root = "/tmp/purple-sandbox";
159        fs::create_dir_all(sandbox_root).map_err(|e| {
160            PurpleError::FilesystemError(format!("Failed to create sandbox root: {}", e))
161        })?;
162
163        // Create necessary directories
164        let directories = [
165            "bin", "lib", "lib64", "usr", "usr/bin", "usr/lib", "tmp", "var", "var/tmp", "proc",
166            "dev", "sys",
167        ];
168
169        for dir in directories.iter() {
170            let path = Path::new(sandbox_root).join(dir);
171            fs::create_dir_all(&path).map_err(|e| {
172                PurpleError::FilesystemError(format!(
173                    "Failed to create directory {}: {}",
174                    path.display(),
175                    e
176                ))
177            })?;
178        }
179
180        // Setup bind mounts for immutable paths
181        for (host_path, sandbox_path) in &self.policy.filesystem.immutable_mounts {
182            let full_sandbox_path = Path::new(sandbox_root).join(
183                sandbox_path
184                    .strip_prefix("/")
185                    .unwrap_or(sandbox_path.as_path()),
186            );
187
188            // Create parent directory if it doesn't exist
189            if let Some(parent) = full_sandbox_path.parent() {
190                fs::create_dir_all(parent).map_err(|e| {
191                    PurpleError::FilesystemError(format!(
192                        "Failed to create parent directory {}: {}",
193                        parent.display(),
194                        e
195                    ))
196                })?;
197            }
198
199            log::info!(
200                "Binding {} to {}",
201                host_path.display(),
202                full_sandbox_path.display()
203            );
204
205            // Bind mount the host path to the sandbox path
206            mount(
207                Some(host_path.as_path()),
208                &full_sandbox_path,
209                None::<&str>,
210                MsFlags::MS_BIND,
211                None::<&str>,
212            )
213            .map_err(|e| {
214                PurpleError::FilesystemError(format!(
215                    "Failed to bind mount {} to {}: {}",
216                    host_path.display(),
217                    full_sandbox_path.display(),
218                    e
219                ))
220            })?;
221
222            // Make it read-only
223            mount(
224                None::<&str>,
225                &full_sandbox_path,
226                None::<&str>,
227                MsFlags::MS_BIND | MsFlags::MS_REMOUNT | MsFlags::MS_RDONLY,
228                None::<&str>,
229            )
230            .map_err(|e| {
231                PurpleError::FilesystemError(format!(
232                    "Failed to remount {} as read-only: {}",
233                    full_sandbox_path.display(),
234                    e
235                ))
236            })?;
237        }
238
239        // Setup scratch directories
240        for scratch_path in &self.policy.filesystem.scratch_dirs {
241            let full_sandbox_path = Path::new(sandbox_root).join(
242                scratch_path
243                    .strip_prefix("/")
244                    .unwrap_or(scratch_path.as_path()),
245            );
246
247            if let Some(parent) = full_sandbox_path.parent() {
248                fs::create_dir_all(parent).map_err(|e| {
249                    PurpleError::FilesystemError(format!(
250                        "Failed to create parent directory {}: {}",
251                        parent.display(),
252                        e
253                    ))
254                })?;
255            }
256
257            fs::create_dir_all(&full_sandbox_path).map_err(|e| {
258                PurpleError::FilesystemError(format!(
259                    "Failed to create scratch directory {}: {}",
260                    full_sandbox_path.display(),
261                    e
262                ))
263            })?;
264        }
265
266        // Setup output directories (writable)
267        for (host_path, sandbox_path) in &self.policy.filesystem.output_mounts {
268            let full_sandbox_path = Path::new(sandbox_root).join(
269                sandbox_path
270                    .strip_prefix("/")
271                    .unwrap_or(sandbox_path.as_path()),
272            );
273
274            if let Some(parent) = full_sandbox_path.parent() {
275                fs::create_dir_all(parent).map_err(|e| {
276                    PurpleError::FilesystemError(format!(
277                        "Failed to create parent directory {}: {}",
278                        parent.display(),
279                        e
280                    ))
281                })?;
282            }
283
284            fs::create_dir_all(&full_sandbox_path).map_err(|e| {
285                PurpleError::FilesystemError(format!(
286                    "Failed to create output directory {}: {}",
287                    full_sandbox_path.display(),
288                    e
289                ))
290            })?;
291
292            // Bind mount output directory
293            mount(
294                Some(host_path.as_path()),
295                &full_sandbox_path,
296                None::<&str>,
297                MsFlags::MS_BIND,
298                None::<&str>,
299            )
300            .map_err(|e| {
301                PurpleError::FilesystemError(format!(
302                    "Failed to bind mount output directory {} to {}: {}",
303                    host_path.display(),
304                    full_sandbox_path.display(),
305                    e
306                ))
307            })?;
308        }
309
310        // Mount essential filesystem
311        // CORRECT: Mount fresh procfs
312        mount(
313            Some("proc"),
314            &Path::new(sandbox_root).join("proc"),
315            Some("proc"),
316            MsFlags::empty(),
317            None::<&str>,
318        )
319        .map_err(|e| PurpleError::FilesystemError(format!("Failed to mount proc: {}", e)))?;
320
321        // WORKAROUND: Bind mount host /dev
322        mount(
323            Some("/dev"),
324            &Path::new(sandbox_root).join("dev"),
325            None::<&str>,
326            MsFlags::MS_BIND | MsFlags::MS_REC,
327            None::<&str>,
328        )
329        .map_err(|e| PurpleError::FilesystemError(format!("Failed to bind mount dev: {}", e)))?;
330
331        // WORKAROUND: Bind mount host /sys
332        mount(
333            Some("/sys"),
334            &Path::new(sandbox_root).join("sys"),
335            None::<&str>,
336            MsFlags::MS_BIND | MsFlags::MS_REC,
337            None::<&str>,
338        )
339        .map_err(|e| PurpleError::FilesystemError(format!("Failed to bind mount sys: {}", e)))?;
340
341        // Setup DNS configuration for network access
342        if !self.policy.network.isolated {
343            let resolv_conf_path = Path::new(sandbox_root).join("etc/resolv.conf");
344            if let Some(parent) = resolv_conf_path.parent() {
345                fs::create_dir_all(parent).map_err(|e| {
346                    PurpleError::FilesystemError(format!(
347                        "Failed to create DNS config directory {}: {}",
348                        parent.display(),
349                        e
350                    ))
351                })?;
352            }
353
354            // Use Google's public DNS servers
355            let resolv_conf_content = "nameserver 8.8.8.8\nnameserver 8.8.4.4\n";
356            fs::write(&resolv_conf_path, resolv_conf_content).map_err(|e| {
357                PurpleError::FilesystemError(format!("Failed to write DNS configuration: {}", e))
358            })?;
359            log::info!("Configured DNS resolvers for network access");
360        }
361
362        // Change root to the sandbox directory
363        log::info!("Changing root to {}", sandbox_root);
364        chroot(sandbox_root).map_err(|e| {
365            PurpleError::FilesystemError(format!("Failed to chroot to {}: {}", sandbox_root, e))
366        })?;
367
368        // Change working directory
369        if let Err(e) = std::env::set_current_dir(&self.policy.filesystem.working_dir) {
370            return Err(PurpleError::FilesystemError(format!(
371                "Failed to change working directory to {}: {}",
372                self.policy.filesystem.working_dir.display(),
373                e
374            )));
375        }
376
377        Ok(())
378    }

392
393        // Store the cgroup manager for cleanup later
394        // In a real implementation, we would keep this and use it in cleanup

463    /// Executes the agent command within the sandbox
464    /// Note: Currently unused but kept for future implementation
465    #[allow(dead_code)]
466    fn execute_agent_command(&self) -> Result<()> {
467        log::info!("Executing agent command within sandbox...");
468
469        if self.agent_command.is_empty() {
470            return Err(PurpleError::CommandError(
471                "No command specified".to_string(),
472            ));
473        }
474
475        let mut command_builder = Command::new(&self.agent_command[0]);
476        if self.agent_command.len() > 1 {
477            command_builder.args(&self.agent_command[1..]);
478        }
479
480        command_builder
481            .stdin(Stdio::inherit())
482            .stdout(Stdio::inherit())
483            .stderr(Stdio::inherit());
484
485        let status = command_builder
486            .status()
487            .map_err(|e| PurpleError::CommandError(format!("Failed to execute command: {}", e)))?;
488
489        if !status.success() {
490            return Err(PurpleError::CommandError(format!(
491                "Command exited with non-zero status: {}",
492                status
493            )));
494        }
495
496        log::info!("Agent command executed successfully");
497        Ok(())
498    }
