10/// Creates a ScmpArgCompare from our internal comparison type
11fn create_arg_compare(arg_index: u32, comparison: &CompiledComparison) -> ScmpArgCompare {
12    let value = get_comparison_value(comparison);
13    let op = get_compare_op(comparison);
14
15    // Create the comparison using the appropriate ScmpCompareOp variant
16    // The third parameter (datum) is the value to compare against
17    ScmpArgCompare::new(arg_index, op, value)
18}
19
20/// Gets the ScmpCompareOp for a given comparison type
21/// Note: libseccomp-rs has inconsistent API where some variants are functions
22fn get_compare_op(comparison: &CompiledComparison) -> ScmpCompareOp {
23    match comparison {
24        CompiledComparison::Equal(_) => ScmpCompareOp::Equal,
25        CompiledComparison::NotEqual(_) => ScmpCompareOp::NotEqual,
26        // libseccomp-rs exposes LessOrEqual but not LessThan directly
27        CompiledComparison::LessThan(_) | CompiledComparison::LessThanOrEqual(_) => {
28            ScmpCompareOp::LessOrEqual
29        }
30        // libseccomp-rs exposes GreaterEqual but not GreaterThan directly
31        CompiledComparison::GreaterThan(_) | CompiledComparison::GreaterThanOrEqual(_) => {
32            ScmpCompareOp::GreaterEqual
33        }
34        // Masked comparisons require a special approach - use MaskedEqual with mask as value
35        // The mask is stored in the value field for masked comparisons
36        CompiledComparison::MaskedEqual { mask, .. } => ScmpCompareOp::MaskedEqual(*mask),
37        // MaskedNotEqual is simulated by MaskedEqual
38        CompiledComparison::MaskedNotEqual { mask, .. } => ScmpCompareOp::MaskedEqual(*mask),
39    }
40}
41
42/// Extracts the comparison value from a CompiledComparison
43fn get_comparison_value(comparison: &CompiledComparison) -> u64 {
44    match comparison {
45        CompiledComparison::Equal(v)
46        | CompiledComparison::NotEqual(v)
47        | CompiledComparison::LessThan(v)
48        | CompiledComparison::LessThanOrEqual(v)
49        | CompiledComparison::GreaterThan(v)
50        | CompiledComparison::GreaterThanOrEqual(v) => *v,
51        CompiledComparison::MaskedEqual { value, .. }
52        | CompiledComparison::MaskedNotEqual { value, .. } => *value,
53    }
54}

56/// Applies seccomp filtering based on the compiled syscall policy
57pub fn apply_seccomp_filter(policy: &CompiledSyscallPolicy) -> Result<()> {
58    log::info!("Applying seccomp syscall filter...");
59
60    // Determine default action based on policy
61    let default_action = if policy.default_deny {
62        ScmpAction::KillProcess
63    } else {
64        ScmpAction::Allow
65    };
66
67    // Create a new seccomp filter context
68    let mut ctx = ScmpFilterContext::new_filter(default_action)?;
69
70    if policy.default_deny {
71        log::info!("Seccomp: Setting default action to DENY (kill process)");
72        log::info!(
73            "Seccomp: Allowing {} syscalls",
74            policy.allowed_syscall_numbers.len()
75        );
76
77        // Add rules for each allowed syscall (basic allow rules)
78        for syscall_num in &policy.allowed_syscall_numbers {
79            let syscall = ScmpSyscall::from(*syscall_num as i32);
80            ctx.add_rule(ScmpAction::Allow, syscall)?;
81            log::debug!("Seccomp: Allowed syscall {}", syscall_num);
82        }
83
84        // Apply advanced rules with conditional filtering
85        // These can further restrict syscalls based on argument values
86        apply_advanced_rules(&mut ctx, &policy.advanced_rules)?;
87    } else {
88        log::info!("Seccomp: Setting default action to ALLOW");
89        log::info!(
90            "Seccomp: Denying {} syscalls",
91            policy.denied_syscall_numbers.len()
92        );
93
94        // In allow-by-default mode, explicitly deny specific syscalls
95        for syscall_num in &policy.denied_syscall_numbers {
96            let syscall = ScmpSyscall::from(*syscall_num as i32);
97            ctx.add_rule(ScmpAction::KillProcess, syscall)?;
98            log::debug!("Seccomp: Denied syscall {}", syscall_num);
99        }
100
101        // Apply advanced rules with conditional filtering in allow-by-default mode
102        apply_advanced_rules(&mut ctx, &policy.advanced_rules)?;
103    }
104
105    // Load the filter into the kernel
106    ctx.load()?;
107
108    if policy.default_deny {
109        log::info!(
110            "Syscall filtering policy enforced with {} allowed syscalls and {} advanced rules",
111            policy.allowed_syscall_numbers.len(),
112            policy.advanced_rules.len()
113        );
114    } else {
115        log::info!(
116            "Syscall filtering policy enforced with {} denied syscalls and {} advanced rules",
117            policy.denied_syscall_numbers.len(),
118            policy.advanced_rules.len()
119        );
120    }
121
122    Ok(())
123}

125/// Applies advanced syscall rules with conditional argument filtering
126fn apply_advanced_rules(
127    ctx: &mut ScmpFilterContext,
128    rules: &[CompiledAdvancedSyscallRule],
129) -> Result<()> {
130    for rule in rules {
131        let syscall = ScmpSyscall::from(rule.syscall_number as i32);
132
133        // Determine action for this rule
134        let action = match rule.action {
135            crate::policy::SyscallAction::Allow => ScmpAction::Allow,
136            crate::policy::SyscallAction::Deny => ScmpAction::KillProcess,
137        };
138
139        if rule.conditions.is_empty() {
140            // No conditions - basic rule
141            ctx.add_rule(action, syscall)?;
142            log::debug!(
143                "Seccomp: Advanced rule for {} (no conditions)",
144                rule.syscall_name
145            );
146        } else {
147            // Build conditional rule using libseccomp's conditional API
148            // Convert our conditions to ScmpArgCompare
149            let comparisons: Vec<ScmpArgCompare> = rule
150                .conditions
151                .iter()
152                .map(|cond| create_arg_compare(cond.arg_index, &cond.comparison))
153                .collect();
154
155            // Handle MaskedNotEqual by inverting the condition
156            // This is a workaround since libseccomp-rs doesn't expose MaskedNotEqual directly
157            if rule.conditions.len() == 1
158                && matches!(
159                    &rule.conditions[0].comparison,
160                    CompiledComparison::MaskedNotEqual { .. }
161                )
162            {
163                log::warn!(
164                    "Seccomp: MaskedNotEqual for {} is simulated with MaskedEqual. \
165                         Consider using 'neq' with specific value for more precise control.",
166                    rule.syscall_name
167                );
168            }
169
170            // Use the conditional rule API
171            // The add_rule_conditional method accepts a slice of conditions
172            ctx.add_rule_conditional(action, syscall, &comparisons)?;
173
174            log::debug!(
175                "Seccomp: Advanced rule for {} with {} conditions",
176                rule.syscall_name,
177                rule.conditions.len()
178            );
179        }
180    }
181
182    Ok(())
183}

185/// Syscall name to number mapping for common syscalls
186/// Note: Kept for potential future dynamic syscall resolution
187#[allow(dead_code)]
188/// Syscall name to number mapping - uses libseccomp's built-in database
189/// This is more reliable than hardcoded mappings and works across different kernel versions
190pub fn get_syscall_number(name: &str) -> Option<i64> {
191    match ScmpSyscall::from_name(name) {
192        Ok(syscall) => Some(i32::from(syscall) as i64),
193        Err(_) => {
194            log::debug!("Failed to resolve syscall name: {}", name);
195            None
196        }
197    }
198}

200/// Converts syscall names to numbers for the policy
201/// Note: Kept for potential future dynamic syscall resolution
202#[allow(dead_code)]
203pub fn resolve_syscall_names(names: &[String]) -> BTreeSet<i64> {
204    let mut numbers = BTreeSet::new();
205    for name in names {
206        if let Some(num) = get_syscall_number(name) {
207            numbers.insert(num);
208        } else {
209            log::warn!("Unknown syscall name: {}", name);
210        }
211    }
212    numbers
