3use crate::error::Result;
4use crate::policy::compiler::CompiledSyscallPolicy;
5use libseccomp::{ScmpAction, ScmpFilterContext, ScmpSyscall};
6use std::collections::BTreeSet;
7
8/// Applies seccomp filtering based on the compiled syscall policy
9pub fn apply_seccomp_filter(policy: &CompiledSyscallPolicy) -> Result<()> {
10    log::info!("Applying seccomp syscall filter...");
11
12    // Determine default action based on policy
13    let default_action = if policy.default_deny {
14        ScmpAction::KillProcess
15    } else {
16        ScmpAction::Allow
17    };
18
19    // Create a new seccomp filter context
20    let mut ctx = ScmpFilterContext::new_filter(default_action)?;
21
22    if policy.default_deny {
23        log::info!("Seccomp: Setting default action to DENY (kill process)");
24        log::info!(
25            "Seccomp: Allowing {} syscalls",
26            policy.allowed_syscall_numbers.len()
27        );
28
29        // Add rules for each allowed syscall
30        for syscall_num in &policy.allowed_syscall_numbers {
31            let syscall = ScmpSyscall::from(*syscall_num as i32);
32            ctx.add_rule(ScmpAction::Allow, syscall)?;
33            log::debug!("Seccomp: Allowed syscall {}", syscall_num);
34        }
35    } else {
36        log::info!("Seccomp: Setting default action to ALLOW");
37        log::info!(
38            "Seccomp: Denying {} syscalls",
39            policy.denied_syscall_numbers.len()
40        );
41
42        // In allow-by-default mode, explicitly deny specific syscalls
43        for syscall_num in &policy.denied_syscall_numbers {
44            let syscall = ScmpSyscall::from(*syscall_num as i32);
45            ctx.add_rule(ScmpAction::KillProcess, syscall)?;
46            log::debug!("Seccomp: Denied syscall {}", syscall_num);
47        }
48    }
49
50    // Load the filter into the kernel
51    ctx.load()?;
52
53    if policy.default_deny {
54        log::info!(
55            "Syscall filtering policy enforced with {} allowed syscalls",
56            policy.allowed_syscall_numbers.len()
57        );
58    } else {
59        log::info!(
60            "Syscall filtering policy enforced with {} denied syscalls",
61            policy.denied_syscall_numbers.len()
62        );
63    }
64
65    Ok(())
66}

68/// Syscall name to number mapping for common syscalls
69/// Note: Kept for potential future dynamic syscall resolution
70#[allow(dead_code)]
71/// Syscall name to number mapping - uses libseccomp's built-in database
72/// This is more reliable than hardcoded mappings and works across different kernel versions
73pub fn get_syscall_number(name: &str) -> Option<i64> {
74    match ScmpSyscall::from_name(name) {
75        Ok(syscall) => Some(i32::from(syscall) as i64),
76        Err(_) => {
77            log::debug!("Failed to resolve syscall name: {}", name);
78            None
79        }
80    }
81}

83/// Converts syscall names to numbers for the policy
84/// Note: Kept for potential future dynamic syscall resolution
85#[allow(dead_code)]
86pub fn resolve_syscall_names(names: &[String]) -> BTreeSet<i64> {
87    let mut numbers = BTreeSet::new();
88    for name in names {
89        if let Some(num) = get_syscall_number(name) {
90            numbers.insert(num);
91        } else {
92            log::warn!("Unknown syscall name: {}", name);
93        }
94    }
95    numbers
96}
