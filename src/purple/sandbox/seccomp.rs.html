8/// Applies seccomp filtering based on the compiled syscall policy
9pub fn apply_seccomp_filter(policy: &CompiledSyscallPolicy) -> Result<()> {
10    log::info!("Applying seccomp syscall filter...");
11
12    // Determine default action based on policy
13    let default_action = if policy.default_deny {
14        ScmpAction::KillProcess
15    } else {
16        ScmpAction::Allow
17    };
18
19    // Create a new seccomp filter context
20    let mut ctx = ScmpFilterContext::new_filter(default_action)?;
21
22    if policy.default_deny {
23        log::info!("Seccomp: Setting default action to DENY (kill process)");
24        log::info!(
25            "Seccomp: Allowing {} syscalls",
26            policy.allowed_syscall_numbers.len()
27        );
28
29        // Add rules for each allowed syscall
30        for syscall_num in &policy.allowed_syscall_numbers {
31            let syscall = ScmpSyscall::from(*syscall_num as i32);
32            ctx.add_rule(ScmpAction::Allow, syscall)?;
33            log::debug!("Seccomp: Allowed syscall {}", syscall_num);
34        }
35    } else {
36        log::info!("Seccomp: Setting default action to ALLOW");
37        log::info!("Seccomp: Denying specific syscalls if any were configured");
38
39        // In allow-by-default mode, we would deny specific syscalls
40        // This is not implemented in the current policy structure
41    }
42
43    // Load the filter into the kernel
44    ctx.load()?;
45
46    log::info!(
47        "Syscall filtering policy enforced with {} allowed syscalls",
48        policy.allowed_syscall_numbers.len()
49    );
50
51    Ok(())
52}

427/// Converts syscall names to numbers for the policy
428/// Note: Kept for potential future dynamic syscall resolution
429#[allow(dead_code)]
430pub fn resolve_syscall_names(names: &[String]) -> BTreeSet<i64> {
431    let mut numbers = BTreeSet::new();
432    for name in names {
433        if let Some(num) = get_syscall_number(name) {
434            numbers.insert(num);
435        } else {
436            log::warn!("Unknown syscall name: {}", name);
437        }
438    }
439    numbers
