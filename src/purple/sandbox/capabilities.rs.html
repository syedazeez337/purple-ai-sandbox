52/// Sets the NO_NEW_PRIVS flag to prevent privilege escalation
53fn prctl_set_no_new_privs() -> Result<()> {
54    // PR_SET_NO_NEW_PRIVS = 38
55    const PR_SET_NO_NEW_PRIVS: libc::c_int = 38;
56
57    let result = unsafe { libc::prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0) };
58    if result != 0 {
59        return Err(PurpleError::CapabilityError(format!(
60            "prctl(PR_SET_NO_NEW_PRIVS) failed with error: {}",
61            std::io::Error::last_os_error()
62        )));
63    }
64    Ok(())
65}

67/// Drops capabilities according to policy
68pub fn drop_capabilities(policy: &CompiledCapabilityPolicy) -> Result<()> {
69    log::info!("Dropping capabilities...");
70
71    if policy.default_drop {
72        log::info!("Capability policy: Drop all capabilities by default");
73
74        // Actual capability dropping implementation using libcap
75        actual_drop_capabilities()?;
76
77        if !policy.added_capabilities.is_empty() {
78            log::info!(
79                "Adding back {} capabilities:",
80                policy.added_capabilities.len()
81            );
82            add_specific_capabilities(&policy.added_capabilities)?;
83        } else {
84            log::info!("No capabilities added back - minimal privilege set");
85        }
86
87        log::info!("Capability management fully configured and enforced");
88    } else {
89        log::info!("Capability policy: Keep all capabilities by default");
90        log::warn!("This is less secure - consider using default_drop=true");
91
92        // Only drop specific capabilities if configured
93        if !policy.dropped_capabilities.is_empty() {
94            drop_specific_capabilities(&policy.dropped_capabilities)?;
95        }
96    }
97
98    log::info!("Capability dropping completed and enforced");
99    Ok(())
100}

113/// Adds specific capabilities back to the process
114fn add_specific_capabilities(capabilities: &HashSet<String>) -> Result<()> {
115    use caps::CapSet;
116
117    log::info!("Adding specific capabilities to process");
118
119    // Get current permitted set first
120    let mut current_permitted = caps::read(None, CapSet::Permitted).map_err(|e| {
121        PurpleError::CapabilityError(format!("Failed to read permitted capabilities: {}", e))
122    })?;
123
124    let mut current_effective = caps::read(None, CapSet::Effective).map_err(|e| {
125        PurpleError::CapabilityError(format!("Failed to read effective capabilities: {}", e))
126    })?;
127
128    for cap_name in capabilities {
129        if let Some(cap) = capability_from_name(cap_name) {
130            log::info!("  - Adding capability: {} ({:?})", cap_name, cap);
131            current_permitted.insert(cap);
132            current_effective.insert(cap);
133        } else {
134            log::warn!("  - Unknown capability: {} (skipping)", cap_name);
135        }
136    }
137
138    // Set the updated capability sets
139    caps::set(None, CapSet::Permitted, &current_permitted).map_err(|e| {
140        PurpleError::CapabilityError(format!("Failed to set permitted capabilities: {}", e))
141    })?;
142
143    caps::set(None, CapSet::Effective, &current_effective).map_err(|e| {
144        PurpleError::CapabilityError(format!("Failed to set effective capabilities: {}", e))
145    })?;
146
147    log::info!("✓ Specific capabilities added and enforced");
148    Ok(())
149}

151/// Drops specific capabilities from the process
152fn drop_specific_capabilities(capabilities: &HashSet<String>) -> Result<()> {
153    use caps::CapSet;
154
155    log::info!("Dropping specific capabilities from process");
156
157    for cap_name in capabilities {
158        if let Some(cap) = capability_from_name(cap_name) {
159            log::info!("  - Dropping capability: {} ({:?})", cap_name, cap);
160
161            // Drop from bounding set
162            if let Err(e) = caps::drop(None, CapSet::Bounding, cap) {
163                log::debug!("Could not drop {:?} from bounding set: {}", cap, e);
164            }
165
166            // Drop from effective set
167            if let Err(e) = caps::drop(None, CapSet::Effective, cap) {
168                log::debug!("Could not drop {:?} from effective set: {}", cap, e);
169            }
170
171            // Drop from permitted set
172            if let Err(e) = caps::drop(None, CapSet::Permitted, cap) {
173                log::debug!("Could not drop {:?} from permitted set: {}", cap, e);
174            }
175
176            // Drop from inheritable set
177            if let Err(e) = caps::drop(None, CapSet::Inheritable, cap) {
178                log::debug!("Could not drop {:?} from inheritable set: {}", cap, e);
179            }
180        } else {
181            log::warn!("  - Unknown capability: {} (skipping)", cap_name);
182        }
183    }
184
185    log::info!("✓ Specific capabilities dropped and enforced");
186    Ok(())
187}

189/// Drop capabilities using system calls
190fn drop_capabilities_system_call() -> Result<()> {
191    use caps::{CapSet, Capability, CapsHashSet};
192
193    log::info!("Dropping all capabilities from process using caps library");
194
195    // Get all capabilities that exist on this system

200        Capability::CAP_FOWNER,

207        Capability::CAP_NET_BIND_SERVICE,
208        Capability::CAP_NET_BROADCAST,
209        Capability::CAP_NET_ADMIN,
210        Capability::CAP_NET_RAW,
211        Capability::CAP_IPC_LOCK,
212        Capability::CAP_IPC_OWNER,
213        Capability::CAP_SYS_MODULE,
214        Capability::CAP_SYS_RAWIO,
215        Capability::CAP_SYS_CHROOT,
216        Capability::CAP_SYS_PTRACE,
217        Capability::CAP_SYS_PACCT,
218        Capability::CAP_SYS_ADMIN,
219        Capability::CAP_SYS_BOOT,
220        Capability::CAP_SYS_NICE,
221        Capability::CAP_SYS_RESOURCE,
222        Capability::CAP_SYS_TIME,
223        Capability::CAP_SYS_TTY_CONFIG,
224        Capability::CAP_MKNOD,
225        Capability::CAP_LEASE,
226        Capability::CAP_AUDIT_WRITE,
227        Capability::CAP_AUDIT_CONTROL,
228        Capability::CAP_SETFCAP,
229        Capability::CAP_MAC_OVERRIDE,
230        Capability::CAP_MAC_ADMIN,
231        Capability::CAP_SYSLOG,
232        Capability::CAP_WAKE_ALARM,
233        Capability::CAP_BLOCK_SUSPEND,
234        Capability::CAP_AUDIT_READ,
235    ];
236
237    // Drop all capabilities from the bounding set
238    log::info!("Dropping capabilities from bounding set...");
239    for cap in &all_caps {
240        if let Err(e) = caps::drop(None, CapSet::Bounding, *cap) {
241            // Some capabilities might not exist on older kernels, that's OK
242            log::debug!(
243                "Could not drop {:?} from bounding set: {} (may not exist)",
244                cap,
245                e
246            );
247        }
248    }
249    log::info!("✓ Bounding set capabilities dropped");
250
251    // Clear the effective capability set
252    log::info!("Clearing effective capability set...");
253    let empty_set: CapsHashSet = CapsHashSet::new();
254    caps::set(None, CapSet::Effective, &empty_set).map_err(|e| {
255        PurpleError::CapabilityError(format!("Failed to clear effective capabilities: {}", e))
256    })?;
257    log::info!("✓ Effective capabilities cleared");
258
259    // Clear the permitted capability set
260    log::info!("Clearing permitted capability set...");
261    caps::set(None, CapSet::Permitted, &empty_set).map_err(|e| {
262        PurpleError::CapabilityError(format!("Failed to clear permitted capabilities: {}", e))
263    })?;
264    log::info!("✓ Permitted capabilities cleared");
265
266    // Clear the inheritable capability set
267    log::info!("Clearing inheritable capability set...");
268    caps::set(None, CapSet::Inheritable, &empty_set).map_err(|e| {
269        PurpleError::CapabilityError(format!("Failed to clear inheritable capabilities: {}", e))
270    })?;
271    log::info!("✓ Inheritable capabilities cleared");
272
273    // Set NO_NEW_PRIVS to prevent privilege escalation via setuid/setgid binaries
274    log::info!("Setting NO_NEW_PRIVS flag...");
275    if let Err(e) = prctl_set_no_new_privs() {
276        log::warn!("Could not set NO_NEW_PRIVS: {} (continuing anyway)", e);
277    } else {
278        log::info!("✓ NO_NEW_PRIVS flag set");
279    }
280
281    log::info!("All capabilities successfully dropped and enforced");
282    Ok(())
283}
