10/// Creates a new user namespace and maps the current user and group IDs.
11///
12/// This is a critical step for isolation, as it allows the sandboxed process to run as
13/// an unprivileged user (e.g., nobody) inside the sandbox, while its parent process
14/// can still manage it.
15///
16/// Returns the new UID and GID for the sandboxed process if successful.
17pub fn unshare_user_namespace() -> Result<(Uid, Gid), String> {
18    // Capture current real user ID (ruid) and group ID (rgid) BEFORE unsharing.
19    // Once we unshare, we are in a new namespace where we don't exist yet (overflow UID),
20    // so we need these original IDs to create the mapping.
21    let ruid = Uid::current();
22    let rgid = Gid::current();
23
24    // Unshare user namespace
25    unshare(CloneFlags::CLONE_NEWUSER)
26        .map_err(|e| {
27            if e == nix::errno::Errno::EINVAL {
28                format!("Failed to unshare user namespace: {}. This can happen if user namespaces are disabled, the process is multi-threaded, or you are already in a restricted nested namespace environment.", e)
29            } else {
30                format!("Failed to unshare user namespace: {}", e)
31            }
32        })?;
33    log::info!("User namespace unshared.");
34
35    // Write "deny" to setgroups FIRST.
36    // This is strictly required before writing to gid_map for unprivileged users,
37    // and good practice to do early.
38    let setgroups_path = PathBuf::from("/proc/self/setgroups");
39    match fs::File::options().write(true).open(&setgroups_path) {
40        Ok(mut setgroups_file) => {
41            if let Err(e) = setgroups_file.write_all(b"deny\n") {
42                return Err(format!(
43                    "Failed to write 'deny' to {}: {}",
44                    setgroups_path.display(),
45                    e
46                ));
47            }
48            log::info!("setgroups denied.");
49        }
50        Err(e) => {
51            // In some container environments (like GitHub Actions runners or Docker with certain profiles),
52            // opening setgroups might fail with EPERM even after unshare.
53            // Check if setgroups is already denied before proceeding
54            if e.kind() == std::io::ErrorKind::PermissionDenied {
55                // Verify setgroups is already denied by reading current value
56                let current = fs::read_to_string(&setgroups_path).unwrap_or_default();
57                if current.trim() == "deny" {
58                    log::info!("setgroups already denied");
59                } else {
60                    return Err(format!(
61                        "Cannot write to {} and it's not already denied. Current value: '{}'. \
62                         This is a security requirement. Error: {}",
63                        setgroups_path.display(),
64                        current.trim(),
65                        e
66                    ));
67                }
68            } else {
69                return Err(format!(
70                    "Failed to open {}: {}",
71                    setgroups_path.display(),
72                    e
73                ));
74            }
75        }
76    }
77
78    // Write UID map
79    let uid_map_path = PathBuf::from("/proc/self/uid_map");
80    match fs::File::options().write(true).open(&uid_map_path) {
81        Ok(mut uid_map_file) => {
82            let uid_map_content = format!("0 {} 1\n", ruid);
83            if let Err(e) = uid_map_file.write_all(uid_map_content.as_bytes()) {
84                return Err(format!(
85                    "Failed to write to {}: {}",
86                    uid_map_path.display(),
87                    e
88                ));
89            } else {
90                log::info!("UID map written: {}", uid_map_content.trim());
91            }
92        }
93        Err(e) => return Err(format!("Failed to open {}: {}", uid_map_path.display(), e)),
94    }
95
96    // Write GID map
97    let gid_map_path = PathBuf::from("/proc/self/gid_map");
98    match fs::File::options().write(true).open(&gid_map_path) {
99        Ok(mut gid_map_file) => {
100            let gid_map_content = format!("0 {} 1\n", rgid);
101            if let Err(e) = gid_map_file.write_all(gid_map_content.as_bytes()) {
102                return Err(format!(
103                    "Failed to write to {}: {}",
104                    gid_map_path.display(),
105                    e
106                ));
107            } else {
108                log::info!("GID map written: {}", gid_map_content.trim());
109            }
110        }
111        Err(e) => return Err(format!("Failed to open {}: {}", gid_map_path.display(), e)),
112    }
113
114    // Set new UIDs/GIDs within the namespace. After mapping,
115    // the process can set its IDs to 0 (root inside new namespace)
116    // without requiring root privileges outside.
117    // Note: setgroups is not allowed because we wrote "deny" to /proc/self/setgroups
118    // setgroups(&[Gid::from_raw(0)]).map_err(|e| format!("Failed to setgroups in new user namespace: {}", e))?;
119
120    setresgid(Gid::from_raw(0), Gid::from_raw(0), Gid::from_raw(0))
121        .map_err(|e| format!("Failed to setresgid in new user namespace: {}", e))?;
122    setresuid(Uid::from_raw(0), Uid::from_raw(0), Uid::from_raw(0))
123        .map_err(|e| format!("Failed to setresuid in new user namespace: {}", e))?;
124    log::info!("User/Group IDs set to 0 (root within new namespace).");
125
126    Ok((Uid::from_raw(0), Gid::from_raw(0)))
127}

129/// Creates a new PID namespace.
130/// Processes inside this namespace will have their own PID 1.
131pub fn unshare_pid_namespace() -> Result<(), String> {
132    unshare(CloneFlags::CLONE_NEWPID)
133        .map_err(|e| format!("Failed to unshare PID namespace: {}", e))?;
134    log::info!("PID namespace unshared.");
135    Ok(())
136}

138/// Creates a new mount namespace.
139/// This allows the sandbox to have its own view of the filesystem mounts.
140/// Uses MS_SHARED propagation so that bind mounts created in the parent
141/// are visible in this child namespace - critical for pivot_root to work.
142pub fn unshare_mount_namespace() -> Result<(), String> {
143    unshare(CloneFlags::CLONE_NEWNS)
144        .map_err(|e| format!("Failed to unshare mount namespace: {}", e))?;
145    log::info!("Mount namespace unshared.");
146
147    // CRITICAL: Use SHARED propagation so parent's bind mounts are visible.
148    // Without this, bind mounts created before fork won't be visible after
149    // we enter the mount namespace, causing "command not found" errors.
150    nix::mount::mount(
151        None::<&Path>,
152        Path::new("/"),
153        None::<&CStr>,
154        nix::mount::MsFlags::MS_REC | nix::mount::MsFlags::MS_SHARED,
155        None::<&CStr>,
156    )
157    .map_err(|e| format!("Failed to make root filesystem shared: {}", e))?;
158    log::info!("Root filesystem set to shared propagation.");
159
160    Ok(())
161}

165pub fn unshare_network_namespace() -> Result<(), String> {
166    unshare(CloneFlags::CLONE_NEWNET)

169    // In a real implementation, you would then set up a veth pair to connect to the host or provide no network.
