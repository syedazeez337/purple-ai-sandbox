46/// Manages resource allocation across sandboxes
47#[derive(Debug, Default)]
48pub struct ResourcePool {
49    pub total_cpu_cores: f64,
50    pub total_memory_mb: u64,
51    pub allocated_cpu: f64,
52    pub allocated_memory: u64,
53}
54
55impl ResourcePool {
56    /// Creates a new resource pool with system resources
57    pub fn new() -> Self {
58        // In a real implementation, this would detect system resources
59        ResourcePool {
60            total_cpu_cores: 4.0,  // Default to 4 cores
61            total_memory_mb: 8192, // Default to 8GB
62            allocated_cpu: 0.0,
63            allocated_memory: 0,
64        }
65    }
66
67    /// Allocates resources for a new sandbox
68    pub fn allocate(&mut self, policy: &CompiledPolicy) -> Result<ResourceAllocation> {
69        // Calculate required resources from policy
70        let required_cpu = policy.resources.cpu_shares.unwrap_or(1.0);
71        let required_memory = policy
72            .resources
73            .memory_limit_bytes
74            .map(|b| b / 1024 / 1024)
75            .unwrap_or(512);
76
77        // Check if resources are available
78        if self.allocated_cpu + required_cpu > self.total_cpu_cores {
79            return Err(PurpleError::ResourceError(format!(
80                "Insufficient CPU resources: {} required, {} available",
81                required_cpu,
82                self.total_cpu_cores - self.allocated_cpu
83            )));
84        }
85
86        if self.allocated_memory + required_memory > self.total_memory_mb {
87            return Err(PurpleError::ResourceError(format!(
88                "Insufficient memory resources: {}MB required, {}MB available",
89                required_memory,
90                self.total_memory_mb - self.allocated_memory
91            )));
92        }
93
94        // Allocate resources
95        self.allocated_cpu += required_cpu;
96        self.allocated_memory += required_memory;
97
98        Ok(ResourceAllocation {
99            cpu_shares: required_cpu,
100            memory_mb: required_memory,
101        })
102    }
103
104    /// Releases resources from a completed sandbox
105    pub fn release(&mut self, allocation: &ResourceAllocation) {
106        self.allocated_cpu -= allocation.cpu_shares;
107        self.allocated_memory -= allocation.memory_mb;
108    }
109}

127    /// Creates a new sandbox with resource allocation
128    pub fn create_sandbox(
129        &mut self,
130        policy: CompiledPolicy,
131        agent_command: Vec<String>,
132    ) -> Result<String> {
133        let mut sandboxes = self.sandboxes.lock().map_err(|e| {
134            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
135        })?;
136
137        // Allocate resources
138        let allocation = self.resource_pool.allocate(&policy)?;
139
140        // Generate unique sandbox ID
141        let sandbox_id = Uuid::new_v4().to_string();
142
143        // Create sandbox
144        let sandbox = Sandbox::new(policy, agent_command);
145
146        // Create sandbox instance
147        let instance = SandboxInstance {
148            sandbox,
149            status: SandboxStatus::Initializing,
150            start_time: std::time::SystemTime::now(),
151            resource_usage: ResourceUsage::default(),
152        };
153
154        // Store in manager
155        sandboxes.insert(sandbox_id.clone(), instance);
156
157        log::info!(
158            "Created sandbox {} with resources: {:.2} CPU, {}MB memory",
159            sandbox_id,
160            allocation.cpu_shares,
161            allocation.memory_mb
162        );
163
164        Ok(sandbox_id)
165    }

167    /// Executes a sandbox and returns the result
168    pub fn execute_sandbox(&self, sandbox_id: &str) -> Result<i32> {
169        let mut sandboxes = self.sandboxes.lock().map_err(|e| {
170            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
171        })?;
172
173        if let Some(instance) = sandboxes.get_mut(sandbox_id) {
174            instance.status = SandboxStatus::Running;
175
176            // Execute the sandbox
177            let exit_code = instance.sandbox.execute()?;
178
179            instance.status = SandboxStatus::Completed;
180
181            log::info!(
182                "Sandbox {} completed with exit code {}",
183                sandbox_id,
184                exit_code
185            );
186
187            Ok(exit_code)
188        } else {
189            Err(PurpleError::SandboxError(format!(
190                "Sandbox {} not found",
191                sandbox_id
192            )))
193        }
194    }

274/// Resource pool status information
275#[derive(Debug)]
276pub struct ResourcePoolStatus {
277    pub total_cpu: f64,
278    pub allocated_cpu: f64,
279    pub total_memory: u64,
280    pub allocated_memory: u64,
281}
282
283impl ResourcePoolStatus {
284    pub fn available_cpu(&self) -> f64 {
285        self.total_cpu - self.allocated_cpu
286    }
287
288    pub fn available_memory(&self) -> u64 {
289        self.total_memory - self.allocated_memory
290    }
291
292    pub fn cpu_utilization(&self) -> f64 {
293        if self.total_cpu > 0.0 {
294            self.allocated_cpu / self.total_cpu
295        } else {
296            0.0
297        }
298    }
299
300    pub fn memory_utilization(&self) -> f64 {
301        if self.total_memory > 0 {
302            self.allocated_memory as f64 / self.total_memory as f64
303        } else {
304            0.0
305        }
306    }
