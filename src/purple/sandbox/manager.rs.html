3/// # Sandbox Manager - Multi-Sandbox Management System
4///
5/// This module provides advanced sandbox management capabilities including:
6/// - Multi-sandbox instance management
7/// - Resource allocation and tracking
8/// - System resource pool management
9/// - Concurrent sandbox execution
10///
11/// ## Current Status
12///
13/// ⚠️  This module is fully implemented and tested but not yet integrated
14/// into the main execution flow. It's currently used in integration tests
15/// (`tests/test_manager.rs`) to verify multi-sandbox functionality.
16///
17/// ## Integration Plan
18///
19/// To integrate this manager into the main application:
20/// 1. Replace direct Sandbox::new() calls with manager.create_sandbox()
21/// 2. Use manager.execute_sandbox() instead of sandbox.execute()
22/// 3. Implement proper resource cleanup via manager.cleanup_sandbox()
23/// 4. Add manager.list_sandboxes() to CLI for monitoring
24///
25/// ## Benefits of Integration
26///
27/// - Prevents resource exhaustion from multiple sandboxes
28/// - Provides centralized monitoring and management
29/// - Enables resource quotas and fair sharing
30/// - Supports concurrent sandbox execution
31///
32/// ## Usage Example
33///
34/// ```rust
35/// let mut manager = SandboxManager::new();
36/// let sandbox_id = manager.create_sandbox(policy, command)?;
37/// let exit_code = manager.execute_sandbox(&sandbox_id)?;
38/// let status = manager.get_sandbox_status(&sandbox_id)?;
39/// manager.cleanup_sandbox(&sandbox_id)?;
40/// ```

88/// Manages resource allocation across sandboxes
89#[derive(Debug, Default)]
90pub struct ResourcePool {
91    pub total_cpu_cores: f64,
92    pub total_memory_mb: u64,
93    pub allocated_cpu: f64,
94    pub allocated_memory: u64,
95}
96
97impl ResourcePool {
98    /// Creates a new resource pool with system resources
99    pub fn new() -> Self {
100        let mut sys = System::new_all();
101        sys.refresh_all();
102
103        let total_cpu = sys.cpus().len() as f64;
104        let total_memory = sys.total_memory() / 1024 / 1024; // Convert to MB
105
106        log::info!(
107            "Detected system resources: {} CPUs, {}MB RAM",
108            total_cpu,
109            total_memory
110        );
111
112        ResourcePool {
113            total_cpu_cores: total_cpu,
114            total_memory_mb: total_memory,
115            allocated_cpu: 0.0,
116            allocated_memory: 0,
117        }
118    }
119
120    /// Allocates resources for a new sandbox
121    pub fn allocate(&mut self, policy: &CompiledPolicy) -> Result<ResourceAllocation> {
122        // Calculate required resources from policy
123        let required_cpu = policy.resources.cpu_shares.unwrap_or(1.0);
124        let required_memory = policy
125            .resources
126            .memory_limit_bytes
127            .map(|b| b / 1024 / 1024)
128            .unwrap_or(512);
129
130        // Check if resources are available
131        if self.allocated_cpu + required_cpu > self.total_cpu_cores {
132            return Err(PurpleError::ResourceError(format!(
133                "Insufficient CPU resources: {} required, {} available",
134                required_cpu,
135                self.total_cpu_cores - self.allocated_cpu
136            )));
137        }
138
139        if self.allocated_memory + required_memory > self.total_memory_mb {
140            return Err(PurpleError::ResourceError(format!(
141                "Insufficient memory resources: {}MB required, {}MB available",
142                required_memory,
143                self.total_memory_mb - self.allocated_memory
144            )));
145        }
146
147        // Allocate resources
148        self.allocated_cpu += required_cpu;
149        self.allocated_memory += required_memory;
150
151        Ok(ResourceAllocation {
152            cpu_shares: required_cpu,
153            memory_mb: required_memory,
154        })
155    }
156
157    /// Releases resources from a completed sandbox
158    pub fn release(&mut self, allocation: &ResourceAllocation) {
159        self.allocated_cpu -= allocation.cpu_shares;
160        self.allocated_memory -= allocation.memory_mb;
161    }
162}

181    /// Creates a new sandbox with resource allocation
182    pub fn create_sandbox(
183        &mut self,
184        policy: CompiledPolicy,
185        agent_command: Vec<String>,
186    ) -> Result<String> {
187        let mut sandboxes = self.sandboxes.lock().map_err(|e| {
188            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
189        })?;
190
191        // Allocate resources
192        let allocation = self.resource_pool.allocate(&policy)?;
193
194        // Generate unique sandbox ID
195        let sandbox_id = Uuid::new_v4().to_string();
196
197        // Create sandbox
198        let sandbox = Sandbox::new(policy, agent_command);
199
200        // Create sandbox instance
201        let instance = SandboxInstance {
202            sandbox,
203            status: SandboxStatus::Initializing,
204            start_time: std::time::SystemTime::now(),
205            resource_usage: ResourceUsage::default(),
206            allocation: allocation.clone(),
207        };
208
209        // Store in manager
210        sandboxes.insert(sandbox_id.clone(), instance);
211
212        log::info!(
213            "Created sandbox {} with resources: {:.2} CPU, {}MB memory",
214            sandbox_id,
215            allocation.cpu_shares,
216            allocation.memory_mb
217        );
218
219        Ok(sandbox_id)
220    }
221
222    /// Executes a sandbox and returns the result
223    pub fn execute_sandbox(&self, sandbox_id: &str) -> Result<i32> {
224        let mut sandboxes = self.sandboxes.lock().map_err(|e| {
225            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
226        })?;
227
228        if let Some(instance) = sandboxes.get_mut(sandbox_id) {
229            instance.status = SandboxStatus::Running;
230
231            // Execute the sandbox
232            let exit_code = instance.sandbox.execute()?;
233
234            instance.status = SandboxStatus::Completed;
235
236            log::info!(
237                "Sandbox {} completed with exit code {}",
238                sandbox_id,
239                exit_code
240            );
241
242            Ok(exit_code)
243        } else {
244            Err(PurpleError::SandboxError(format!(
245                "Sandbox {} not found",
246                sandbox_id
247            )))
248        }
249    }

267    /// Lists all active sandboxes
268    pub fn list_sandboxes(&self) -> Result<Vec<(String, SandboxStatus)>> {
269        let sandboxes = self.sandboxes.lock().map_err(|e| {
270            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
271        })?;
272
273        Ok(sandboxes
274            .iter()
275            .map(|(id, instance)| (id.clone(), instance.status.clone()))
276            .collect())
277    }

279    /// Cleans up a completed sandbox
280    pub fn cleanup_sandbox(&mut self, sandbox_id: &str) -> Result<()> {
281        let mut sandboxes = self.sandboxes.lock().map_err(|e| {
282            PurpleError::SandboxError(format!("Failed to lock sandbox manager: {}", e))
283        })?;
284
285        if let Some(instance) = sandboxes.remove(sandbox_id) {
286            // Perform actual cleanup of resources
287            log::info!("Cleaning up resources for sandbox {}", sandbox_id);
288
289            // 1. Clean up filesystem, cgroups, etc.
290            if let Err(e) = instance.sandbox.cleanup_and_audit() {
291                log::warn!("Error during sandbox cleanup: {}", e);
292                // Continue with resource release even if cleanup failed
293            }
294
295            // 2. Release allocated resources back to the pool
296            self.resource_pool.release(&instance.allocation);
297
298            log::info!("Cleaned up sandbox {}", sandbox_id);
299            Ok(())
300        } else {
301            Err(PurpleError::SandboxError(format!(
302                "Sandbox {} not found",
303                sandbox_id
304            )))
305        }
306    }

334    /// Creates a new sandbox from profile and command strings (CLI helper)
335    pub fn create_sandbox_from_profile(
336        &mut self,
337        name: String,
338        profile_name: String,
339    ) -> Result<String> {
340        // Load policy from file
341        let policy_file = format!("./policies/{}.yaml", profile_name);
342        let policy =
343            crate::policy::parser::load_policy_from_file(std::path::Path::new(&policy_file))?
344                .compile()?;
345
346        // Use profile's command or default to echo
347        let command = vec![
348            "/bin/sh".to_string(),
349            "-c".to_string(),
350            format!(
351                "echo 'Sandbox {} created with profile {}'",
352                name, profile_name
353            ),
354        ];
355
356        self.create_sandbox(policy, command)
357    }

360/// Resource pool status information
361#[derive(Debug)]
362#[allow(dead_code)]
363pub struct ResourcePoolStatus {
364    pub total_cpu: f64,
365    pub allocated_cpu: f64,
366    pub total_memory: u64,
367    pub allocated_memory: u64,
368}
369
370#[allow(dead_code)]
371impl ResourcePoolStatus {
372    pub fn available_cpu(&self) -> f64 {
373        self.total_cpu - self.allocated_cpu
374    }
375
376    pub fn available_memory(&self) -> u64 {
377        self.total_memory - self.allocated_memory
378    }
379
380    pub fn cpu_utilization(&self) -> f64 {
381        if self.total_cpu > 0.0 {
382            self.allocated_cpu / self.total_cpu
383        } else {
384            0.0
385        }
386    }
387
388    pub fn memory_utilization(&self) -> f64 {
389        if self.total_memory > 0 {
390            self.allocated_memory as f64 / self.total_memory as f64
391        } else {
392            0.0
393        }
394    }
395}
