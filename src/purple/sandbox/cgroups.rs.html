7/// Manages cgroups for resource limit enforcement
8pub struct CgroupManager {
9    cgroup_name: String,

11}

13impl CgroupManager {
14    /// Creates a new cgroup manager for the sandbox
15    pub fn new(sandbox_id: &str) -> Self {
16        let cgroup_name = format!("purple-sandbox-{}", sandbox_id);
17        let cgroup_path = PathBuf::from(format!("/sys/fs/cgroup/purple/{}", cgroup_name));
18
19        CgroupManager {
20            cgroup_name,
21            cgroup_path,
22        }
23    }

25    /// Sets up cgroups and applies resource limits
26    pub fn setup_cgroups(&self, policy: &CompiledResourcePolicy) -> Result<()> {
27        log::info!("Setting up cgroups for resource limits...");
28
29        use std::fs;
30
31        log::info!("Cgroup name: {}", self.cgroup_name);
32        log::info!("Cgroup path: {}", self.cgroup_path.display());
33
34        // Create the cgroup directory
35        if let Err(e) = fs::create_dir_all(&self.cgroup_path) {
36            if e.kind() == std::io::ErrorKind::PermissionDenied
37                || e.kind() == std::io::ErrorKind::ReadOnlyFilesystem
38            {
39                log::warn!(
40                    "Failed to create cgroup directory: {}. Resource limits will NOT be enforced.",
41                    e
42                );
43                return Ok(());
44            }
45            return Err(crate::error::PurpleError::ResourceError(format!(
46                "Failed to create cgroup directory: {}",
47                e
48            )));
49        }
50
51        // Apply CPU limits
52        if let Some(cpu_shares) = policy.cpu_shares {
53            log::info!("Setting CPU shares to {}", cpu_shares);
54            // Convert shares to appropriate value (cgroup2 uses different scale)
55            let cpu_shares_value = (cpu_shares * 1024.0) as u64;
56            let cpu_shares_path = self.cgroup_path.join("cpu.max");
57            fs::write(cpu_shares_path, format!("{}", cpu_shares_value))?;
58        } else {
59            log::info!("No CPU shares limit specified");
60        }
61
62        // Apply memory limits
63        if let Some(memory_limit) = policy.memory_limit_bytes {
64            log::info!(
65                "Setting memory limit to {} bytes ({} MB)",
66                memory_limit,
67                memory_limit / 1024 / 1024
68            );
69            let memory_limit_path = self.cgroup_path.join("memory.max");
70            fs::write(memory_limit_path, format!("{}", memory_limit))?;
71        } else {
72            log::info!("No memory limit specified");
73        }
74
75        // Apply process limits
76        if let Some(pids_limit) = policy.pids_limit {
77            log::info!("Setting process limit to {}", pids_limit);
78            let pids_limit_path = self.cgroup_path.join("pids.max");
79            fs::write(pids_limit_path, format!("{}", pids_limit))?;
80        } else {
81            log::info!("No process limit specified");
82        }
83
84        // Apply I/O limits (cgroup2 uses different approach)
85        if let Some(io_limit) = policy.block_io_limit_bytes_per_sec {
86            log::info!(
87                "Setting I/O limit to {} bytes/sec ({} MB/s)",
88                io_limit,
89                io_limit / 1024 / 1024
90            );
91            // For cgroup2, we would configure io.max or io.bfq.weight
92            // This is more complex and may require root privileges
93        } else {
94            log::info!("No I/O limit specified");
95        }
96
97        // Apply timeout limits
98        if let Some(timeout_secs) = policy.session_timeout_seconds {
99            log::info!("Setting session timeout to {} seconds", timeout_secs);
100            // Would implement process monitoring and termination
101        } else {
102            log::info!("No session timeout specified");
103        }
104
105        log::info!("Cgroups configured using cgroupfs");
106        Ok(())
107    }

133/// Creates a unique sandbox ID for cgroup naming
134pub fn generate_sandbox_id() -> String {
135    use std::time::{SystemTime, UNIX_EPOCH};
136
137    match SystemTime::now().duration_since(UNIX_EPOCH) {
138        Ok(duration) => format!("{}", duration.as_secs()),
139        Err(_) => "0".to_string(),
140    }
