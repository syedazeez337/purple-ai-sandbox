28    /// Detect the root device major:minor numbers dynamically
29    /// Returns (major, minor) or None if detection fails
30    fn detect_root_device() -> Option<(u64, u64)> {
31        // Try to read from /proc/self/mountinfo first
32        if let Ok(file) = fs::File::open("/proc/self/mountinfo") {
33            let reader = BufReader::new(file);
34            for line in reader.lines().map_while(|r| r.ok()) {
35                // Look for the root filesystem entry
36                if line.contains(" / ") {
37                    // Parse format: <id> <parent> <major>:<minor> <root> <mountpoint> <options>
38                    let parts: Vec<&str> = line.split_whitespace().collect();
39                    if parts.len() >= 4 {
40                        let device_parts: Vec<&str> = parts[2].split(':').collect();
41                        if device_parts.len() == 2
42                            && let (Ok(major), Ok(minor)) = (
43                                device_parts[0].parse::<u64>(),
44                                device_parts[1].parse::<u64>(),
45                            )
46                        {
47                            return Some((major, minor));
48                        }
49                    }
50                }
51            }
52        }
53
54        // Fallback: try to detect common devices
55        // Check if /dev/sda exists (common for traditional disks)
56        if Path::new("/dev/sda").exists() {
57            return Some((8, 0)); // Traditional SCSI/SATA device
58        }
59
60        // Check if /dev/nvme0n1 exists (common for NVMe)
61        if Path::new("/dev/nvme0n1").exists() {
62            return Some((259, 0)); // NVMe device
63        }
64
65        // Check if /dev/vda exists (common for virtual machines)
66        if Path::new("/dev/vda").exists() {
67            return Some((253, 0)); // Virtio device
68        }
69
70        // Default fallback to traditional device (8:0)
71        Some((8, 0))
72    }

74    /// Sets up cgroups and applies resource limits
75    pub fn setup_cgroups(&self, policy: &CompiledResourcePolicy) -> Result<()> {
76        log::info!("Setting up cgroups for resource limits...");
77
78        use std::fs;
79
80        log::info!("Cgroup name: {}", self.cgroup_name);
81        log::info!("Cgroup path: {}", self.cgroup_path.display());
82
83        // Ensure parent cgroup directory exists and has controllers enabled
84        let parent_cgroup_path = config::cgroup_parent_path();
85        if !parent_cgroup_path.exists() {
86            if let Err(e) = fs::create_dir_all(&parent_cgroup_path) {
87                return Err(crate::error::PurpleError::ResourceError(format!(
88                    "Failed to create parent cgroup directory: {}",
89                    e
90                )));
91            }
92            log::info!(
93                "Created parent cgroup directory: {}",
94                parent_cgroup_path.display()
95            );
96        }
97
98        // Enable controllers in parent cgroup's subtree_control
99        // This is required for child cgroups to have controller files (cpu.max, memory.max, etc.)
100        let subtree_control_path = parent_cgroup_path.join("cgroup.subtree_control");
101        if subtree_control_path.exists() {
102            // Read current controllers
103            let current_controllers = fs::read_to_string(&subtree_control_path).unwrap_or_default();
104            let needs_cpu = !current_controllers.contains("cpu");
105            let needs_memory = !current_controllers.contains("memory");
106            let needs_pids = !current_controllers.contains("pids");
107            let needs_io = !current_controllers.contains("io");
108
109            if needs_cpu || needs_memory || needs_pids || needs_io {
110                log::info!("Enabling cgroup controllers in parent subtree_control...");
111
112                // Build list of controllers to enable
113                let mut controllers_to_enable = Vec::new();
114                if needs_cpu {
115                    controllers_to_enable.push("+cpu");
116                }
117                if needs_memory {
118                    controllers_to_enable.push("+memory");
119                }
120                if needs_pids {
121                    controllers_to_enable.push("+pids");
122                }
123                if needs_io {
124                    controllers_to_enable.push("+io");
125                }
126
127                // Enable each controller individually (some systems require this)
128                for controller in &controllers_to_enable {
129                    if let Err(e) = fs::write(&subtree_control_path, controller) {
130                        log::warn!(
131                            "Failed to enable controller {} in subtree_control: {}",
132                            controller,
133                            e
134                        );
135                    } else {
136                        log::info!("Enabled controller {} in parent cgroup", controller);
137                    }
138                }
139            } else {
140                log::info!("All required cgroup controllers already enabled");
141            }
142        }
143
144        // Create the cgroup directory
145        if let Err(e) = fs::create_dir_all(&self.cgroup_path) {
146            if e.kind() == std::io::ErrorKind::PermissionDenied
147                || e.kind() == std::io::ErrorKind::ReadOnlyFilesystem
148            {
149                log::error!(
150                    "CRITICAL: Failed to create cgroup directory due to permissions: {}. This means resource limits CANNOT be enforced. The sandbox will continue but resource isolation is COMPROMISED.",
151                    e
152                );
153                return Err(crate::error::PurpleError::ResourceError(format!(
154                    "Cgroup creation failed: {}. Resource limits cannot be enforced - sandbox would be insecure.",
155                    e
156                )));
157            }
158            return Err(crate::error::PurpleError::ResourceError(format!(
159                "Failed to create cgroup directory: {}",
160                e
161            )));
162        }
163
164        // Apply CPU limits
165        if let Some(cpu_shares) = policy.cpu_shares {
166            log::info!("Setting CPU shares to {}", cpu_shares);
167            // cgroup2 cpu.max format: "$MAX $PERIOD" where MAX is quota in microseconds
168            // and PERIOD is the period (typically 100000 = 100ms)
169            // For example, 0.5 shares = 50% CPU = "50000 100000"
170            let period: u64 = 100000; // 100ms period
171            let quota = (cpu_shares * period as f64) as u64;
172            let cpu_max_path = self.cgroup_path.join("cpu.max");
173            fs::write(&cpu_max_path, format!("{} {}", quota, period))?;
174            log::info!(
175                "CPU limit set: {} us / {} us period ({}%)",
176                quota,
177                period,
178                (cpu_shares * 100.0) as u32
179            );
180        } else {
181            log::info!("No CPU shares limit specified");
182        }
183
184        // Apply memory limits
185        if let Some(memory_limit) = policy.memory_limit_bytes {
186            log::info!(
187                "Setting memory limit to {} bytes ({} MB)",
188                memory_limit,
189                memory_limit / 1024 / 1024
190            );
191            let memory_limit_path = self.cgroup_path.join("memory.max");
192            fs::write(memory_limit_path, format!("{}", memory_limit))?;
193
194            // Disable swap to ensure strict enforcement
195            let swap_limit_path = self.cgroup_path.join("memory.swap.max");
196            if swap_limit_path.exists() {
197                if let Err(e) = fs::write(swap_limit_path, "0") {
198                    log::warn!("Failed to disable swap (non-fatal): {}", e);
199                } else {
200                    log::info!("Swap disabled for strict memory enforcement");
201                }
202            }
203        } else {
204            log::info!("No memory limit specified");
205        }
206
207        // Apply process limits
208        if let Some(pids_limit) = policy.pids_limit {
209            log::info!("Setting process limit to {}", pids_limit);
210            let pids_limit_path = self.cgroup_path.join("pids.max");
211            fs::write(pids_limit_path, format!("{}", pids_limit))?;
212        } else {
213            log::info!("No process limit specified");
214        }
215
216        // Apply I/O limits using cgroup2 io.max interface
217        if let Some(io_limit) = policy.block_io_limit_bytes_per_sec {
218            log::info!(
219                "Setting I/O limit to {} bytes/sec ({} MB/s)",
220                io_limit,
221                io_limit / 1024 / 1024
222            );
223
224            // For cgroup2, use io.max to set I/O rate limits
225            // Format: "<major>:<minor> <type> <limit>"
226            // Detect root device dynamically
227            if let Some((major, minor)) = Self::detect_root_device() {
228                let io_max_path = self.cgroup_path.join("io.max");
229                let io_max_content = format!(
230                    "{}:{} rbytes={} wbytes={}",
231                    major, minor, io_limit, io_limit
232                );
233
234                match fs::write(io_max_path, io_max_content) {
235                    Ok(_) => log::info!("✓ I/O rate limit applied successfully"),
236                    Err(e) => {
237                        log::warn!(
238                            "Failed to set I/O rate limit: {}. This may require root privileges.",
239                            e
240                        );
241                        log::warn!("I/O rate limiting will not be enforced.");
242                    }
243                }
244            } else {
245                log::warn!("Failed to detect root device, I/O limits will not be applied");
246            }
247        } else {
248            log::info!("No I/O limit specified");
249        }
250
251        // Apply timeout limits
252        if let Some(timeout_secs) = policy.session_timeout_seconds {
253            log::info!("Setting session timeout to {} seconds", timeout_secs);
254            // Note: Actual timeout enforcement happens in the sandbox execution
255            // through the parent's wait loop with timeout checking
256        } else {
257            log::info!("No session timeout specified");
258        }
259
260        log::info!("Cgroups configured using cgroupfs");
261        Ok(())
262    }

287    /// Validates that cgroup functionality is available and working
288    pub fn validate_cgroup_functionality() -> Result<()> {
289        use std::fs;
290
291        log::info!("Validating cgroup functionality...");
292
293        // Check if cgroup filesystem is mounted
294        let cgroup_mounts = ["/sys/fs/cgroup", "/sys/fs/cgroup/unified"];
295
296        let mut found_cgroup = false;
297        for mount_point in &cgroup_mounts {
298            if Path::new(mount_point).exists() {
299                found_cgroup = true;
300                log::info!("Found cgroup mount at: {}", mount_point);
301                break;
302            }
303        }
304
305        if !found_cgroup {
306            return Err(crate::error::PurpleError::ResourceError(
307                "Cgroup filesystem not found. Resource limits cannot be enforced.".to_string(),
308            ));
309        }
310
311        // Check if our purple cgroup parent exists
312        let purple_cgroup_path = config::cgroup_parent_path();
313        if !purple_cgroup_path.exists() {
314            log::info!(
315                "Creating purple cgroup parent directory: {}",
316                purple_cgroup_path.display()
317            );
318            if let Err(e) = fs::create_dir_all(&purple_cgroup_path) {
319                return Err(crate::error::PurpleError::ResourceError(format!(
320                    "Failed to create purple cgroup directory: {}",
321                    e
322                )));
323            }
324        }
325
326        // Test creating a temporary cgroup directory
327        let test_cgroup_path = purple_cgroup_path.join("test_validation");
328        match fs::create_dir_all(&test_cgroup_path) {
329            Ok(_) => {
330                // Successfully created, now clean up
331                if let Err(e) = fs::remove_dir(&test_cgroup_path) {
332                    log::warn!("Failed to clean up test cgroup: {}", e);
333                }
334                log::info!("✓ Cgroup functionality validated successfully");
335                Ok(())
336            }
337            Err(e) => {
338                if e.kind() == std::io::ErrorKind::PermissionDenied {
339                    return Err(crate::error::PurpleError::ResourceError(format!(
340                        "Insufficient permissions to create cgroups: {}. Run as root or configure proper cgroup permissions.",
341                        e
342                    )));
343                }
344                Err(crate::error::PurpleError::ResourceError(format!(
345                    "Cgroup validation failed: {}",
346                    e
347                )))
348            }
349        }
350    }

352    /// Cleans up cgroups after sandbox execution
353    pub fn cleanup_cgroups(&self) -> Result<()> {
354        use std::fs;
355
356        log::info!("Cleaning up cgroups for: {}", self.cgroup_name);
357
358        // First, terminate any remaining processes in the cgroup
359        self.kill_cgroup_processes()?;
360
361        // Wait a moment for processes to terminate
362        std::thread::sleep(std::time::Duration::from_millis(100));
363
364        // Remove the cgroup directory with retries
365        if self.cgroup_path.exists() {
366            let max_retries = 5;
367            let mut last_error = None;
368
369            for attempt in 1..=max_retries {
370                match fs::remove_dir(&self.cgroup_path) {
371                    Ok(_) => {
372                        log::info!("✓ Cgroup {} removed successfully", self.cgroup_name);
373                        return Ok(());
374                    }
375                    Err(e) => {
376                        // EBUSY means processes are still running or kernel is cleaning up
377                        if e.kind() == std::io::ErrorKind::Other || e.raw_os_error() == Some(16) {
378                            if attempt == 1 {
379                                log::warn!(
380                                    "Cgroup {} still busy, attempting force cleanup...",
381                                    self.cgroup_name
382                                );
383                                // Try to force kill if not already done
384                                self.force_kill_cgroup_processes()?;
385                            }
386                            log::debug!("Attempt {}/{} failed: {}", attempt, max_retries, e);
387                            std::thread::sleep(std::time::Duration::from_millis(
388                                200 * attempt as u64,
389                            ));
390                            last_error = Some(e);
391                        } else if e.kind() == std::io::ErrorKind::NotFound {
392                            log::debug!("Cgroup {} already removed", self.cgroup_name);
393                            return Ok(());
394                        } else {
395                            // Other errors are likely fatal/permanent
396                            log::warn!("Failed to remove cgroup {}: {}", self.cgroup_name, e);
397                            return Err(crate::error::PurpleError::ResourceError(format!(
398                                "Failed to remove cgroup: {}",
399                                e
400                            )));
401                        }
402                    }
403                }
404            }
405
406            if let Some(e) = last_error {
407                log::warn!(
408                    "Failed to remove cgroup directory after {} attempts: {}",
409                    max_retries,
410                    e
411                );
412                // Don't fail the sandbox execution for cleanup failure, but log it clearly
413            }
414        } else {
415            log::debug!(
416                "Cgroup directory {} does not exist (already cleaned up)",
417                self.cgroup_path.display()
418            );
419        }
420
421        log::info!("✓ Cgroups cleanup completed");
422        Ok(())
423    }

425    /// Kills all processes in the cgroup
426    fn kill_cgroup_processes(&self) -> Result<()> {
427        use std::fs;
428
429        let cgroup_procs_path = self.cgroup_path.join("cgroup.procs");
430
431        if !cgroup_procs_path.exists() {
432            return Ok(());
433        }
434
435        let procs_content = match fs::read_to_string(&cgroup_procs_path) {
436            Ok(content) => content,
437            Err(e) => {
438                log::debug!("Could not read cgroup.procs: {}", e);
439                return Ok(());
440            }
441        };
442
443        for line in procs_content.lines() {
444            if let Ok(pid) = line.trim().parse::<i32>()
445                && pid > 0
446            {
447                log::debug!("Sending SIGTERM to process {} in cgroup", pid);
448                // Use SIGTERM first for graceful shutdown
449                // SAFETY: libc::kill is safe to call with any PID - returns -1 on error.
450                // The PID was validated to be > 0 and parsed from cgroup.procs.
451                unsafe {
452                    libc::kill(pid, libc::SIGTERM);
453                }
454            }
455        }
456
457        Ok(())
458    }

501    /// Cleans up orphaned cgroups from previous failed runs
502    pub fn cleanup_orphaned_cgroups() -> Result<()> {
503        use std::fs;
504
505        log::info!("Cleaning up orphaned cgroups...");
506
507        let purple_cgroup_path = config::cgroup_parent_path();
508        if !purple_cgroup_path.exists() {
509            log::info!("No purple cgroups found - nothing to clean up");
510            return Ok(());
511        }
512
513        // Read all purple cgroup directories
514        let mut cleaned_count = 0;
515        let mut permission_errors = Vec::new();
516
517        if let Ok(entries) = fs::read_dir(&purple_cgroup_path) {
518            for entry in entries.flatten() {
519                if let Ok(file_type) = entry.file_type()
520                    && file_type.is_dir()
521                {
522                    let cgroup_name = entry.file_name();
523                    let cgroup_name_str = cgroup_name.to_string_lossy();
524
525                    // Skip validation cgroup if it exists
526                    if cgroup_name_str == "test_validation" {
527                        continue;
528                    }
529
530                    let cgroup_path = entry.path();
531
532                    // Check if cgroup has any processes
533                    let cgroup_procs_path = cgroup_path.join("cgroup.procs");
534
535                    // Try to read processes in the cgroup
536                    let has_processes =
537                        if let Ok(procs_content) = fs::read_to_string(&cgroup_procs_path) {
538                            !procs_content.trim().is_empty()
539                        } else {
540                            false
541                        };
542
543                    if has_processes {
544                        log::warn!(
545                            "Found orphaned cgroup with active processes: {}",
546                            cgroup_name_str
547                        );
548                        // Don't clean up cgroups with active processes
549                        continue;
550                    }
551
552                    // Safe to clean up - no processes
553                    match fs::remove_dir(&cgroup_path) {
554                        Ok(_) => {
555                            log::info!("Cleaned up orphaned cgroup: {}", cgroup_name_str);
556                            cleaned_count += 1;
557                        }
558                        Err(e) => {
559                            if e.kind() == std::io::ErrorKind::PermissionDenied {
560                                permission_errors.push(cgroup_path.clone());
561                            } else {
562                                log::warn!(
563                                    "Failed to clean up orphaned cgroup {}: {}",
564                                    cgroup_name_str,
565                                    e
566                                );
567                            }
568                        }
569                    }
570                }
571            }
572        }
573
574        // Provide helpful instructions if there were permission errors
575        if !permission_errors.is_empty() {
576            log::warn!("========================================");
577            log::warn!("ORPHANED CGROUPS DETECTED");
578            log::warn!("========================================");
579            log::warn!("Some cgroups could not be cleaned up due to permissions.");
580            log::warn!("This typically happens when running with sudo vs without.");
581
582            // Try to clean up with sudo if available
583            log::info!("Attempting cleanup with elevated permissions...");
584
585            let mut sudo_cleaned = 0;
586            for cgroup_path in &permission_errors {
587                // Use sudo to remove the cgroup if we can
588                let output = std::process::Command::new("sudo")
589                    .args(["rmdir", cgroup_path.to_str().unwrap_or("")])
590                    .output();
591
592                match output {
593                    Ok(output) if output.status.success() => {
594                        log::info!("Cleaned up cgroup with sudo: {}", cgroup_path.display());
595                        sudo_cleaned += 1;
596                    }
597                    _ => {
598                        log::warn!("Could not clean up: {}", cgroup_path.display());
599                        log::warn!("To fix manually, run: sudo rmdir {}", cgroup_path.display());
600                    }
601                }
602            }
603
604            cleaned_count += sudo_cleaned;
605        }
606
607        log::info!(
608            "Orphaned cgroup cleanup completed. Cleaned {} cgroups.",
609            cleaned_count
610        );
611        Ok(())
612    }

614    /// Reads CPU usage statistics from cgroup
615    pub fn get_cpu_stats(&self) -> Result<f64> {
616        let cpu_stat_path = self.cgroup_path.join("cpu.stat");
617
618        let content = fs::read_to_string(&cpu_stat_path).map_err(|e| {
619            crate::error::PurpleError::ResourceError(format!("Failed to read cpu.stat: {}", e))
620        })?;
621
622        // Parse usage_usec value
623        for line in content.lines() {
624            if line.starts_with("usage_usec") {
625                let parts: Vec<&str> = line.split_whitespace().collect();
626                if parts.len() == 2 {
627                    let usec: u64 = parts[1].parse().map_err(|e| {
628                        crate::error::PurpleError::ResourceError(format!(
629                            "Failed to parse cpu usage: {}",
630                            e
631                        ))
632                    })?;
633                    return Ok(usec as f64 / 1_000_000.0); // Convert to seconds
634                }
635            }
636        }
637
638        Err(crate::error::PurpleError::ResourceError(
639            "usage_usec not found in cpu.stat".to_string(),
640        ))
641    }

656    /// Reads I/O statistics from cgroup
657    pub fn get_io_stats(&self) -> Result<u64> {
658        let io_stat_path = self.cgroup_path.join("io.stat");
659
660        let content: String = fs::read_to_string(&io_stat_path).map_err(|e| {
661            crate::error::PurpleError::ResourceError(format!("Failed to read io.stat: {}", e))
662        })?;
663
664        let mut total_bytes = 0u64;
665
666        // Parse rbytes and wbytes for all devices
667        for line in content.lines() {
668            let parts: Vec<&str> = line.split_whitespace().collect();
669            for part in parts.iter() {
670                if (part.starts_with("rbytes=") || part.starts_with("wbytes="))
671                    && let Some(eq_pos) = part.find('=')
672                {
673                    let value_str = &part[eq_pos + 1..];
674                    if let Ok(bytes) = value_str.parse::<u64>() {
675                        total_bytes += bytes;
676                    }
677                }
678            }
679        }
680
681        Ok(total_bytes)
682    }
683}

685/// Creates a unique sandbox ID for cgroup naming
686pub fn generate_sandbox_id() -> String {
687    use std::time::{SystemTime, UNIX_EPOCH};
688
689    match SystemTime::now().duration_since(UNIX_EPOCH) {
690        Ok(duration) => format!("{}", duration.as_nanos()),
691        Err(_) => "0".to_string(),
692    }
