18fn main() {
19    let cli = Cli::parse();
20
21    // Initialize logging based on CLI argument
22    let log_level = match cli.log_level.to_lowercase().as_str() {
23        "trace" => LevelFilter::Trace,
24        "debug" => LevelFilter::Debug,
25        "info" => LevelFilter::Info,
26        "warn" => LevelFilter::Warn,
27        "error" => LevelFilter::Error,
28        _ => {
29            eprintln!(
30                "Invalid log level '{}', defaulting to 'info'",
31                cli.log_level
32            );
33            LevelFilter::Info
34        }
35    };
36
37    if let Err(e) = init_logging(log_level) {
38        eprintln!("Failed to initialize logging: {}", e);
39        std::process::exit(1);
40    }
41
42    log::info!("Purple sandbox started with log level: {}", cli.log_level);
43
44    match &cli.command {
45        Commands::Init => {
46            println!("Initializing Purple environment...");
47            // TODO: Implement environment initialization logic
48        }
49        Commands::Profile { command } => match command {
50            ProfileCommands::Create { name } => {
51                println!("Creating profile: {}", name);
52                let policy_path = format!("./policies/{}.yaml", name);
53                let policy_path_buf = std::path::PathBuf::from(&policy_path);
54                match policy::parser::load_policy_from_file(&policy_path_buf) {
55                    Ok(policy) => {
56                        println!("Successfully loaded policy for {}: {:?}", name, policy);
57                        match policy.compile() {
58                            Ok(compiled_policy) => {
59                                println!(
60                                    "Successfully compiled policy for {}: {:?}",
61                                    name, compiled_policy
62                                );
63                                // TODO: Implement actual profile creation logic (e.g., save compiled policy)
64                            }
65                            Err(e) => {
66                                eprintln!("Error compiling policy for {}: {}", name, e);
67                            }
68                        }
69                    }
70                    Err(e) => {
71                        eprintln!("Error loading policy for {}: {}", name, e);
72                    }
73                }
74            }
75            ProfileCommands::List => {
76                println!("Listing available profiles...");
77
78                // List YAML files in the policies directory
79                let policies_dir = std::path::Path::new("./policies");
80                if !policies_dir.exists() {
81                    println!("No policies directory found. Creating one...");
82                    if let Err(e) = std::fs::create_dir_all(policies_dir) {
83                        eprintln!("Failed to create policies directory: {}", e);
84                        return;
85                    }
86                    println!("Created policies directory at {}", policies_dir.display());
87                    return;
88                }
89
90                match std::fs::read_dir(policies_dir) {
91                    Ok(entries) => {
92                        let mut profiles = Vec::new();
93
94                        for entry in entries.flatten() {
95                            if let Some(file_name) = entry.file_name().to_str()
96                                && (file_name.ends_with(".yaml") || file_name.ends_with(".yml"))
97                            {
98                                let profile_name = file_name
99                                    .trim_end_matches(".yaml")
100                                    .trim_end_matches(".yml")
101                                    .to_string();
102                                profiles.push(profile_name);
103                            }
104                        }
105
106                        if profiles.is_empty() {
107                            println!("No profiles found in {}", policies_dir.display());
108                            println!("Create a new profile with: purple profile create <name>");
109                        } else {
110                            println!("Available profiles:");
111                            for (i, profile) in profiles.iter().enumerate() {
112                                println!("  {}. {}", i + 1, profile);
113                            }
114                            println!("\nTotal: {} profile(s)", profiles.len());
115                        }
116                    }
117                    Err(e) => {
118                        eprintln!("Error reading policies directory: {}", e);
119                    }
120                }
121            }
122            ProfileCommands::Delete { name } => {
123                println!("Deleting profile: {}", name);
124
125                // Check if profile exists
126                let policy_path = format!("./policies/{}.yaml", name);
127                let policy_path_buf = std::path::PathBuf::from(&policy_path);
128
129                if !policy_path_buf.exists() {
130                    eprintln!("Profile '{}' does not exist", name);
131                    return;
132                }
133
134                // Delete the policy file
135                match std::fs::remove_file(&policy_path_buf) {
136                    Ok(_) => {
137                        println!("Successfully deleted profile: {}", name);
138                    }
139                    Err(e) => {
140                        eprintln!("Error deleting profile '{}': {}", name, e);
141                    }
142                }
143            }
144            ProfileCommands::Show { name } => {
145                println!("Showing profile: {}", name);
146
147                let policy_path = format!("./policies/{}.yaml", name);
148                let policy_path_buf = std::path::PathBuf::from(&policy_path);
149
150                if !policy_path_buf.exists() {
151                    eprintln!("Profile '{}' does not exist", name);
152                    return;
153                }
154
155                match policy::parser::load_policy_from_file(&policy_path_buf) {
156                    Ok(policy) => {
157                        println!("Profile: {}", policy.name);
158                        if let Some(desc) = policy.description {
159                            println!("Description: {}", desc);
160                        }
161
162                        println!("\nFilesystem Policy:");
163                        println!(
164                            "  Immutable paths: {}",
165                            policy.filesystem.immutable_paths.len()
166                        );
167                        println!("  Scratch paths: {}", policy.filesystem.scratch_paths.len());
168                        println!("  Output paths: {}", policy.filesystem.output_paths.len());
169                        println!(
170                            "  Working directory: {}",
171                            policy.filesystem.working_dir.display()
172                        );
173
174                        println!("\nSyscall Policy:");
175                        println!("  Default deny: {}", policy.syscalls.default_deny);
176                        println!("  Allowed syscalls: {}", policy.syscalls.allow.len());
177                        println!("  Denied syscalls: {}", policy.syscalls.deny.len());
178
179                        println!("\nResource Policy:");
180                        if let Some(cpu) = policy.resources.cpu_shares {
181                            println!("  CPU shares: {}", cpu);
182                        }
183                        if let Some(mem) = policy.resources.memory_limit_bytes {
184                            println!("  Memory limit: {}", mem);
185                        }
186                        if let Some(pids) = policy.resources.pids_limit {
187                            println!("  Process limit: {}", pids);
188                        }
189                        if let Some(timeout) = policy.resources.session_timeout_seconds {
190                            println!("  Timeout: {}s", timeout);
191                        }
192
193                        println!("\nCapability Policy:");
194                        println!("  Default drop: {}", policy.capabilities.default_drop);
195                        println!("  Added capabilities: {}", policy.capabilities.add.len());
196
197                        println!("\nNetwork Policy:");
198                        println!("  Isolated: {}", policy.network.isolated);
199                        println!("  Outgoing ports: {}", policy.network.allow_outgoing.len());
200                        println!("  Incoming ports: {}", policy.network.allow_incoming.len());
201
202                        println!("\nAudit Policy:");
203                        println!("  Enabled: {}", policy.audit.enabled);
204                        println!("  Log path: {}", policy.audit.log_path.display());
205                        println!("  Detail levels: {}", policy.audit.detail_level.len());
206                    }
207                    Err(e) => {
208                        eprintln!("Error loading profile '{}': {}", name, e);
209                    }
210                }
211            }
212        },
213        Commands::Run(args) => {
214            let agent_command = args.command.clone(); // Clone to pass into Sandbox::new
215            let profile_name = args.profile.clone();
216
217            println!(
218                "Running AI agent with profile '{}' and command: {:?}",
219                profile_name, agent_command
220            );
221
222            // For now, load/compile policy from example file for run command as well.
223            // In future, this would load a saved compiled policy for the profile.
224            let policy_path = format!("./policies/{}.yaml", profile_name);
225            let policy_path_buf = std::path::PathBuf::from(&policy_path);
226            match policy::parser::load_policy_from_file(&policy_path_buf) {
227                Ok(policy) => {
228                    match policy.compile() {
229                        Ok(compiled_policy) => {
230                            let sandbox = Sandbox::new(compiled_policy, agent_command);
231                            match sandbox.execute() {
232                                Ok(exit_code) => {
233                                    if exit_code != 0 {
234                                        std::process::exit(exit_code);
235                                    }
236                                }
237                                Err(e) => {
238                                    eprintln!("Sandbox execution failed: {}", e);
239                                    eprintln!("\n=== Debugging Information ===");
240
241                                    // Provide context-specific debugging help
242                                    match e {
243                                        PurpleError::SandboxError(ref msg) => {
244                                            eprintln!("Sandbox setup error: {}", msg);
245                                            eprintln!("Possible causes:");
246                                            eprintln!(
247                                                "  - Insufficient permissions for namespace operations"
248                                            );
249                                            eprintln!(
250                                                "  - Missing kernel support for user namespaces"
251                                            );
252                                            eprintln!("  - Filesystem permissions issues");
253                                            eprintln!("\nTry running with:");
254                                            eprintln!(
255                                                "  sudo sysctl -w kernel.unprivileged_userns_clone=1"
256                                            );
257                                        }
258                                        PurpleError::FilesystemError(ref msg) => {
259                                            eprintln!("Filesystem error: {}", msg);
260                                            eprintln!("Possible causes:");
261                                            eprintln!(
262                                                "  - Missing directories or permission issues"
263                                            );
264                                            eprintln!(
265                                                "  - Bind mount failures due to insufficient privileges"
266                                            );
267                                            eprintln!("  - Disk space or inode limitations");
268                                            eprintln!("\nCheck directory permissions and try:");
269                                            eprintln!("  sudo mkdir -p /tmp/purple-sandbox");
270                                            eprintln!("  sudo chmod 777 /tmp/purple-sandbox");
271                                        }
272                                        PurpleError::ResourceError(ref msg) => {
273                                            eprintln!("Resource limit error: {}", msg);
274                                            eprintln!("Possible causes:");
275                                            eprintln!(
276                                                "  - Insufficient permissions to create cgroups"
277                                            );
278                                            eprintln!("  - Cgroup filesystem not mounted");
279                                            eprintln!("  - System resource limits reached");
280                                            eprintln!("\nCheck cgroup setup:");
281                                            eprintln!("  mount | grep cgroup");
282                                            eprintln!(
283                                                "  sudo mount -t cgroup2 none /sys/fs/cgroup"
284                                            );
285                                        }
286                                        PurpleError::SyscallError(ref msg) => {
287                                            eprintln!("Syscall filtering error: {}", msg);
288                                            eprintln!("Possible causes:");
289                                            eprintln!("  - Missing libseccomp library");
290                                            eprintln!("  - Invalid syscall names in policy");
291                                            eprintln!("  - Kernel seccomp support disabled");
292                                            eprintln!("\nInstall required packages:");
293                                            eprintln!("  sudo dnf install libseccomp-devel");
294                                        }
295                                        PurpleError::PolicyError(ref msg) => {
296                                            eprintln!("Policy configuration error: {}", msg);
297                                            eprintln!("Possible causes:");
298                                            eprintln!("  - Invalid YAML syntax in policy file");
299                                            eprintln!("  - Unknown syscall or capability names");
300                                            eprintln!("  - Missing required policy fields");
301                                            eprintln!("\nValidate policy with:");
302                                            eprintln!("  ./purple profile show {}", profile_name);
303                                        }
304                                        _ => {
305                                            eprintln!("Unexpected error type: {:?}", e);
306                                            eprintln!(
307                                                "Please report this issue with the full error message."
308                                            );
309                                        }
310                                    }
311
312                                    std::process::exit(1);
313                                }
314                            }
315                        }
316                        Err(e) => {
317                            eprintln!("Error compiling policy for {}: {}", profile_name, e);
318                            eprintln!("\n=== Policy Compilation Debugging ===");
319                            eprintln!("Possible causes:");
320                            eprintln!("  - Invalid YAML syntax in policy file");
321                            eprintln!(
322                                "  - Unknown syscall names (check src/sandbox/seccomp.rs for supported syscalls)"
323                            );
324                            eprintln!(
325                                "  - Unknown capability names (must match Linux capability names)"
326                            );
327                            eprintln!("  - Missing required fields in policy");
328                            eprintln!("\nPolicy file location: policies/{}.yaml", profile_name);
329                            eprintln!("Validate with: ./purple profile show {}", profile_name);
330                            eprintln!(
331                                "Supported syscalls: read, write, openat, close, fstat, newfstatat, mmap, mprotect, munmap, brk, access, execve, arch_prctl, set_tid_address, set_robust_list, rseq, prlimit64, getrandom, exit_group, clone3"
332                            );
333                            eprintln!(
334                                "Supported deny syscalls: mount, unmount, reboot, kexec_load, bpf, unlinkat, renameat2"
335                            );
336                            std::process::exit(1);
337                        }
338                    }
339                }
340                Err(e) => {
341                    eprintln!("Error loading policy for {}: {}", profile_name, e);
342                    std::process::exit(1);
343                }
344            }
345        }
346    }
