9/// Top-level policy configuration for an AI agent's execution.
10#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
11pub struct Policy {
12    pub name: String,
13    pub description: Option<String>,
14    pub filesystem: FilesystemPolicy,
15    pub syscalls: SyscallPolicy,
16    pub resources: ResourcePolicy,
17    pub capabilities: CapabilityPolicy,
18    pub network: NetworkPolicy,
19    pub audit: AuditPolicy,
20    /// AI-specific policies (optional)
21    #[serde(default)]
22    pub ai_policy: Option<crate::ai::AIPolicies>,
23    /// eBPF monitoring configuration (optional)
24    #[serde(default)]
25    pub ebpf_monitoring: EbpfMonitoringPolicy,
26}

28/// Defines filesystem access rules.
29#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
30pub struct FilesystemPolicy {
31    /// List of paths that are immutable (read-only) inside the sandbox.
32    /// These are typically bind-mounted from the host.
33    #[serde(default)]
34    pub immutable_paths: Vec<PathMapping>,
35    /// List of paths that are scratch (writable, disposable) inside the sandbox.
36    /// These are typically temporary directories created for the sandbox session.
37    #[serde(default)]
38    pub scratch_paths: Vec<PathBuf>,
39    /// List of paths that are output (write-only, review-gated) inside the sandbox.
40    /// These are typically bind-mounted from the host with write-only permissions if possible,
41    /// or marked for special handling by the orchestrator.
42    #[serde(default)]
43    pub output_paths: Vec<PathMapping>,
44    /// Specifies the working directory for the sandboxed process.
45    pub working_dir: PathBuf,
46}

49#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
50pub struct PathMapping {
51    pub host_path: PathBuf,

55/// Defines allowed and denied syscalls.
56#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
57pub struct SyscallPolicy {
58    /// If true, all syscalls are denied by default unless explicitly allowed.
59    /// This should always be true based on the axioms.
60    pub default_deny: bool,
61    /// List of syscalls to explicitly allow.
62    #[serde(default)]
63    pub allow: Vec<String>,
64    /// List of syscalls to explicitly deny (overrides allow if conflict).
65    #[serde(default)]
66    pub deny: Vec<String>,
67    /// Advanced syscall rules with argument filtering.
68    /// These rules override simple allow/deny rules when they conflict.
69    #[serde(default)]
70    pub advanced_rules: Vec<AdvancedSyscallRule>,
71}
72
73/// Advanced syscall rule with argument filtering support.
74#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
75pub struct AdvancedSyscallRule {
76    /// Name of the syscall this rule applies to.
77    pub syscall: String,
78    /// Action to take for this syscall (allow or deny).
79    pub action: SyscallAction,
80    /// Conditions that must be met for this rule to apply.
81    /// If no conditions are specified, the rule always applies.
82    #[serde(default)]
83    pub conditions: Vec<SyscallCondition>,
84}
85
86/// Action to take for a syscall rule.
87#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
88pub enum SyscallAction {
89    #[serde(rename = "allow")]
90    Allow,
91    #[serde(rename = "deny")]
92    Deny,
93}
94
95/// Condition that must be met for a syscall rule to apply.
96#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
97pub struct SyscallCondition {
98    /// Argument index (0-based) to check.
99    /// For x86_64: 0=rax, 1=rdi, 2=rsi, 3=rdx, 4=r10, 5=r8, 6=r9
100    pub arg: usize,
101    /// Comparison operation to perform.
102    pub op: ConditionOp,
103    /// Value to compare against.
104    pub value: u64,
105    /// Optional bitmask for masked comparisons (used with masked_eq).
106    /// The argument value is ANDed with this mask before comparison.
107    #[serde(default)]
108    pub mask: Option<u64>,
109}
110
111/// Comparison operation for syscall argument conditions.
112#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
113pub enum ConditionOp {
114    /// Equal (==)
115    #[serde(rename = "eq")]
116    Equal,
117    /// Not equal (!=)
118    #[serde(rename = "neq")]
119    NotEqual,
120    /// Less than (<)
121    #[serde(rename = "lt")]
122    LessThan,
123    /// Less than or equal (<=)
124    #[serde(rename = "lte")]
125    LessThanOrEqual,
126    /// Greater than (>)
127    #[serde(rename = "gt")]
128    GreaterThan,
129    /// Greater than or equal (>=)
130    #[serde(rename = "gte")]
131    GreaterThanOrEqual,
132    /// Masked equal - value AND mask must equal comparison value
133    #[serde(rename = "masked_eq")]
134    MaskedEqual,
135    /// Masked not equal - value AND mask must not equal comparison value
136    #[serde(rename = "masked_neq")]
137    MaskedNotEqual,
138}

142pub struct ResourcePolicy {

155/// Defines Linux capabilities to be granted or dropped.
156#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
157pub struct CapabilityPolicy {
158    /// If true, all capabilities are dropped by default unless explicitly added.
159    /// This should always be true based on the axioms.
160    pub default_drop: bool,
161    /// List of capabilities to explicitly add (e.g., "CAP_NET_RAW").
162    #[serde(default)]
163    pub add: Vec<String>,
164    /// List of capabilities to drop when default_drop is false.
165    #[serde(default)]
166    pub drop: Vec<String>,
167}

169/// Defines network access rules.
170#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
171pub struct NetworkPolicy {
172    /// If true, the sandbox will have no network access (e.g., isolated network namespace).
173    pub isolated: bool,
174    /// Allowed outgoing connections (e.g., "8.8.8.8:53").
175    #[serde(default)]
176    pub allow_outgoing: Vec<String>,
177    /// Allowed incoming connections.
178    #[serde(default)]
179    pub allow_incoming: Vec<String>,
180    /// Blocked IP addresses (for eBPF filter).
181    #[serde(default)]
182    pub blocked_ips: Vec<String>,
183    /// Custom DNS servers (e.g., ["8.8.8.8", "1.1.1.1"]).
184    #[serde(default)]
185    pub dns_servers: Option<Vec<String>>,
186}
