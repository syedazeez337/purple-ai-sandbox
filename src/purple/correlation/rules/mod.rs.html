17/// Detection rules engine
18#[derive(Debug, Clone)]
19pub struct RulesEngine {
20    rules: Arc<Mutex<Vec<DetectionRule>>>,
21    sigma_rules: Arc<Mutex<Vec<SigmaRule>>>,
22    compiled_rules: Arc<Mutex<HashMap<String, CompiledRule>>>,
23    enabled: bool,
24}

26unsafe impl Send for RulesEngine {}
27unsafe impl Sync for RulesEngine {}

29impl Default for RulesEngine {
30    fn default() -> Self {
31        Self::new(true)
32    }
33}

35impl RulesEngine {
36    pub fn new(enabled: bool) -> Self {
37        Self {
38            rules: Arc::new(Mutex::new(Vec::new())),
39            sigma_rules: Arc::new(Mutex::new(Vec::new())),
40            compiled_rules: Arc::new(Mutex::new(HashMap::new())),
41            enabled,
42        }
43    }

45    /// Load rules from files
46    pub fn load_rules_from_directory(&self, dir: PathBuf) -> Result<(), String> {
47        if !dir.exists() {
48            return Err(format!("Rules directory does not exist: {:?}", dir));
49        }
50
51        let mut loaded_count = 0;
52
53        for entry in WalkDir::new(dir)
54            .follow_links(true)
55            .into_iter()
56            .filter_map(|e| e.ok())
57            .filter(|e| e.file_type().is_file())
58        {
59            if let Some(ext) = entry.path().extension()
60                && (ext == "yaml" || ext == "yml")
61            {
62                match self.load_rule_file(entry.path()) {
63                    Ok(_) => loaded_count += 1,
64                    Err(e) => {
65                        log::warn!("Failed to load rule {}: {}", entry.path().display(), e)
66                    }
67                }
68            }
69        }
70
71        log::info!("Loaded {} detection rules", loaded_count);
72        Ok(())
73    }
74
75    /// Load a single rule file
76    fn load_rule_file(&self, path: &Path) -> Result<(), String> {
77        let content = std::fs::read_to_string(path)
78            .map_err(|e| format!("Failed to read rule file: {}", e))?;
79
80        // Try to parse as Sigma rule first
81        if let Ok(sigma_rule) = serde_yaml::from_str::<SigmaRule>(&content) {
82            let mut sigma_rules = self.sigma_rules.lock().unwrap_or_else(|e| e.into_inner());
83            sigma_rules.push(sigma_rule);
84            return Ok(());
85        }
86
87        // Try to parse as custom detection rule
88        if let Ok(rule) = serde_yaml::from_str::<DetectionRule>(&content) {
89            let mut rules = self.rules.lock().unwrap_or_else(|e| e.into_inner());
90            rules.push(rule);
91            return Ok(());
92        }
93
94        Err("Failed to parse rule file".to_string())
95    }
96
97    /// Add a custom rule
98    pub fn add_rule(&self, rule: DetectionRule) {
99        {
100            let mut rules = self.rules.lock().unwrap_or_else(|e| e.into_inner());
101            rules.push(rule);
102        }
103        self.recompile_rules();
104    }
105
106    /// Compile rules for performance
107    fn recompile_rules(&self) {
108        let mut compiled = self
109            .compiled_rules
110            .lock()
111            .unwrap_or_else(|e| e.into_inner());
112        let rules = self.rules.lock().unwrap_or_else(|e| e.into_inner());
113
114        for rule in rules.iter() {
115            if rule.enabled {
116                let compiled_rule = self.compile_rule(rule);
117                compiled.insert(rule.id.clone(), compiled_rule);
118            }
119        }
120    }
121
122    /// Compile a single rule
123    fn compile_rule(&self, rule: &DetectionRule) -> CompiledRule {
124        let mut compiled = CompiledRule {
125            rule_id: rule.id.clone(),
126            name: rule.name.clone(),
127            severity: rule.severity.clone(),
128            conditions: Vec::new(),
129        };
130
131        #[allow(clippy::regex_creation_in_loops)]
132        for cond in &rule.conditions {
133            let compiled_cond = match cond.operator {
134                ConditionOperator::Contains => CompiledCondition::Contains(
135                    cond.field.clone(),
136                    self.extract_string_value(&cond.value),
137                ),
138                ConditionOperator::Matches => CompiledCondition::Matches(
139                    cond.field.clone(),
140                    Regex::new(&self.extract_string_value(&cond.value))
141                        .unwrap_or_else(|_| Regex::new(".*").unwrap()),
142                ),
143                ConditionOperator::Equals => {
144                    CompiledCondition::Equals(cond.field.clone(), self.extract_value(&cond.value))
145                }
146                ConditionOperator::GreaterThan => CompiledCondition::GreaterThan(
147                    cond.field.clone(),
148                    self.extract_f64(&cond.value).unwrap_or(0.0),
149                ),
150                _ => CompiledCondition::Equals(cond.field.clone(), cond.value.clone()),
151            };
152            compiled.conditions.push(compiled_cond);
153        }
154
155        compiled
156    }
157
158    /// Match an event against all rules
159    pub fn match_event(&self, event: &EnrichedEvent) -> Vec<RuleMatch> {
160        if !self.enabled {
161            return Vec::new();
162        }
163
164        let mut matches = Vec::new();
165        let compiled_rules = self
166            .compiled_rules
167            .lock()
168            .unwrap_or_else(|e| e.into_inner());
169
170        for (_id, rule) in compiled_rules.iter() {
171            if self.evaluate_rule(rule, event) {
172                matches.push(RuleMatch {
173                    rule_id: rule.rule_id.clone(),
174                    rule_name: rule.name.clone(),
175                    severity: rule.severity.clone(),
176                    matched_conditions: rule.conditions.len(),
177                });
178            }
179        }
180
181        matches
182    }
183
184    /// Evaluate a compiled rule against an event
185    fn evaluate_rule(&self, rule: &CompiledRule, event: &EnrichedEvent) -> bool {
186        for cond in &rule.conditions {
187            if !self.evaluate_condition(cond, event) {
188                return false;
189            }
190        }
191        true
192    }
193
194    /// Evaluate a condition against an event
195    fn evaluate_condition(&self, cond: &CompiledCondition, event: &EnrichedEvent) -> bool {
196        let (_field, result) = match cond {
197            CompiledCondition::Equals(field, expected) => {
198                let value = self.get_event_value(field, event);
199                (field.clone(), value == *expected)
200            }
201            CompiledCondition::Contains(field, substring) => {
202                let value = self.get_event_value(field, event);
203                (field.clone(), value.to_string().contains(substring))
204            }
205            CompiledCondition::Matches(field, regex) => {
206                let value = self.get_event_value(field, event);
207                (field.clone(), regex.is_match(&value.to_string()))
208            }
209            CompiledCondition::GreaterThan(field, threshold) => {
210                let value = self.get_event_value(field, event);
211                (
212                    field.clone(),
213                    self.extract_f64(&value)
214                        .map(|v| v > *threshold)
215                        .unwrap_or(false),
216                )
217            }
218            CompiledCondition::LessThan(field, threshold) => {
219                let value = self.get_event_value(field, event);
220                (
221                    field.clone(),
222                    self.extract_f64(&value)
223                        .map(|v| v < *threshold)
224                        .unwrap_or(false),
225                )
226            }
227        };
228
229        result
230    }
231
232    /// Get a value from an event by field path
233    fn get_event_value(&self, field: &str, event: &EnrichedEvent) -> serde_json::Value {
234        match field {
235            "event_type" => serde_json::json!(event.base.event_type),
236            "severity" => serde_json::json!(format!("{:?}", event.severity)),
237            "category" => serde_json::json!(format!("{:?}", event.base.category)),
238            "pid" => serde_json::json!(event.base.pid),
239            "comm" => serde_json::json!(event.base.comm),
240            "details" => serde_json::json!(event.base.details),
241            "risk_score" => serde_json::json!(event.risk_score),
242            "is_expected" => serde_json::json!(event.is_expected),
243            _ => serde_json::json!(null),
244        }
245    }
246
247    fn extract_string_value(&self, value: &serde_json::Value) -> String {
248        value.as_str().unwrap_or("").to_string()
249    }
250
251    fn extract_f64(&self, value: &serde_json::Value) -> Option<f64> {
252        value
253            .as_f64()
254            .or(value.as_u64().map(|u| u as f64))
255            .or(value.as_i64().map(|i| i as f64))
256    }
257
258    fn extract_value(&self, value: &serde_json::Value) -> serde_json::Value {
259        value.clone()
260    }
261
262    /// Get rule count
263    pub fn get_rule_count(&self) -> usize {
264        let rules = self.rules.lock().unwrap_or_else(|e| e.into_inner());
265        let sigma_rules = self.sigma_rules.lock().unwrap_or_else(|e| e.into_inner());
266        rules.len() + sigma_rules.len()
267    }
268
269    /// Get all rules
270    pub fn get_all_rules(&self) -> Vec<DetectionRule> {
271        let rules = self.rules.lock().unwrap_or_else(|e| e.into_inner());
272        rules.clone()
273    }

285#[derive(Debug, Clone)]

329#[derive(Debug, Clone, Serialize, Deserialize)]
330pub struct SigmaDetection {
331    pub selection: Option<HashMap<String, serde_yaml::Value>>,
332    pub filter: Option<HashMap<String, serde_yaml::Value>>,
333    pub condition: Option<String>,
334}
335
336impl SigmaRule {
337    /// Convert Sigma rule to internal DetectionRule format
338    pub fn to_detection_rule(&self) -> Option<DetectionRule> {
339        let detection = self.detection.as_ref()?;
340        let selection = detection.selection.as_ref()?;
341        let _condition = detection.condition.as_ref()?;
342
343        let mut conditions = Vec::new();
344
345        for (field, value) in selection {
346            let operator = match value {
347                serde_yaml::Value::String(s) if s.starts_with("*") => ConditionOperator::Contains,
348                serde_yaml::Value::String(_) => ConditionOperator::Equals,
349                serde_yaml::Value::Number(_) => ConditionOperator::GreaterThan,
350                _ => ConditionOperator::Equals,
351            };
352
353            conditions.push(RuleCondition {
354                field: field.clone(),
355                operator,
356                value: serde_json::to_value(value).ok()?,
357            });
358        }
359
360        let severity = match self.level.as_deref() {
361            Some("critical") => Severity::Critical,
362            Some("high") => Severity::High,
363            Some("medium") => Severity::Medium,
364            Some("low") => Severity::Low,
365            _ => Severity::Medium,
366        };
367
368        Some(DetectionRule {
369            id: self.id.clone(),
370            name: self.title.clone(),
371            description: self.description.clone().unwrap_or_default(),
372            severity,
373            conditions,
374            actions: vec![RuleAction::Alert],
375            enabled: self.status.as_deref() != Some("deprecated"),
376            tags: self.tags.clone().unwrap_or_default(),
377        })
378    }

484    fn test_rule_engine() {
485        let engine = RulesEngine::new(true);
486
487        // Add a test rule
488        let rule = DetectionRule {
489            id: "test_001".to_string(),
490            name: "Test Rule".to_string(),
491            description: "Test rule for unit testing".to_string(),
492            severity: Severity::High,
493            conditions: vec![RuleCondition {
494                field: "category".to_string(),
495                operator: ConditionOperator::Equals,
496                value: serde_json::json!("FileAccess"),
497            }],
498            actions: vec![RuleAction::Alert],
499            enabled: true,
500            tags: vec!["test".to_string()],
501        };
502
503        engine.add_rule(rule);
504
505        // Test matching
506        let event = EnrichedEvent {
507            base: RawEvent::new(
508                "syscall".to_string(),
509                1234,
510                "openat file".to_string(),
511                EventCategory::FileAccess,

513            ..Default::default()

517        assert_eq!(matches.len(), 1);
